C++ — 智能指针示例 
#include <iostream>
#include <memory>
#include <vector>
using namespace std;

class Resource {
public:
    string name;
    Resource(string n) : name(n) {
        cout << "[构造] " << name << endl;
    }
    ~Resource() {
        cout << "[析构] " << name << endl;
    }
};

int main() {
    // unique_ptr：独占所有权
    auto p1 = make_unique<Resource>("独占资源");

    // shared_ptr：共享所有权，引用计数
    auto s1 = make_shared<Resource>("共享资源");
    auto s2 = s1; // 引用计数 → 2
    cout << "引用计数: " << s1.use_count() << endl;

    // 离开作用域，自动释放内存
    return 0;
} // 无需手动 delete！
