🔄 1. 迭代器基础

STL 算法通过迭代器操作容器，迭代器的分类决定了算法的能力与复杂度。理解迭代器类别是正确使用算法的前提。

  • 输入迭代器（Input）：只读、单次遍历，如 istream_iterator
  • 输出迭代器（Output）：只写、单次遍历，如 ostream_iterator
  • 前向迭代器（Forward）：可读写、多次遍历，如 forward_list
  • 双向迭代器（Bidirectional）：支持 --，如 list、set
  • 随机访问迭代器（Random Access）：支持 +n、[]，如 vector、deque
迭代器基本用法 
#include <algorithm>
#include <vector>
#include <list>
#include <iterator>
#include <iostream>

int main() {
    std::vector<int> v = {3, 1, 4, 1, 5, 9, 2, 6};

    // begin/end 返回迭代器
    auto first = v.begin();  // 指向第一个元素
    auto last  = v.end();    // 指向最后一个元素的下一位（past-the-end）

    // 迭代器运算（随机访问迭代器专有）
    auto mid = v.begin() + v.size() / 2;
    std::cout << "中间元素: " << *mid << "\n";

    // std::distance：两个迭代器之间的距离（对前向迭代器为 O(n)）
    std::cout << "元素数量: " << std::distance(first, last) << "\n";

    // std::advance：迭代器前进 n 步
    auto it = v.begin();
    std::advance(it, 3);
    std::cout << "第 4 个元素: " << *it << "\n";

    // 反向迭代器
    for (auto rit = v.rbegin(); rit != v.rend(); ++rit)
        std::cout << *rit << " ";
    std::cout << "\n";

    // 范围 for（本质上使用 begin/end）
    for (int x : v)
        std::cout << x << " ";
    std::cout << "\n";

    return 0;
}
💡

头文件约定：几乎所有 STL 算法都在 <algorithm> 中，数值算法（accumulate、iota 等）在 <numeric> 中。

📊 2. 排序与查找

sort 是最常用的算法，要求随机访问迭代器，平均复杂度 O(n log n)。binary_search 等二分算法要求序列已排序。

sort / stable_sort / find / binary_search 
#include <algorithm>
#include <vector>
#include <string>
#include <iostream>

int main() {
    std::vector<int> v = {5, 3, 8, 1, 9, 2, 7, 4, 6};

    // sort：不稳定排序，O(n log n)
    std::sort(v.begin(), v.end());
    // v = {1, 2, 3, 4, 5, 6, 7, 8, 9}

    // stable_sort：保留相等元素的原始相对顺序（对象排序常用）
    struct Student { std::string name; int score; };
    std::vector<Student> students = {{"Alice",90},{"Bob",85},{"Cara",90},{"Dan",85}};
    std::stable_sort(students.begin(), students.end(),
        [](const Student& a, const Student& b){ return a.score > b.score; });
    // score 相同时，Alice 仍在 Cara 前，Bob 仍在 Dan 前

    // find：线性查找，返回指向第一个匹配元素的迭代器
    auto it = std::find(v.begin(), v.end(), 5);
    if (it != v.end())
        std::cout << "找到 5，位于下标 " << std::distance(v.begin(), it) << "\n";

    // find_if：条件查找
    auto even_it = std::find_if(v.begin(), v.end(),
        [](int x){ return x % 2 == 0; });
    if (even_it != v.end())
        std::cout << "第一个偶数: " << *even_it << "\n";

    // find_if_not：查找第一个不满足条件的元素
    auto odd_it = std::find_if_not(v.begin(), v.end(),
        [](int x){ return x % 2 == 0; });
    std::cout << "第一个奇数: " << *odd_it << "\n";

    // binary_search：有序序列的二分查找，返回 bool（O(log n)）
    bool found = std::binary_search(v.begin(), v.end(), 7);
    std::cout << "binary_search(7): " << std::boolalpha << found << "\n";

    // lower_bound / upper_bound：返回插入位置的迭代器
    auto lb = std::lower_bound(v.begin(), v.end(), 5);  // >= 5 的第一个
    auto ub = std::upper_bound(v.begin(), v.end(), 5);  // > 5 的第一个
    std::cout << "lower_bound(5) = " << *lb << ", upper_bound(5) = " << *ub << "\n";

    // equal_range：同时返回 lower/upper_bound（常用于关联容器）
    auto [lo, hi] = std::equal_range(v.begin(), v.end(), 5);
    std::cout << "5 的个数: " << std::distance(lo, hi) << "\n";

    return 0;
}
⚠️

注意：binary_search、lower_bound、upper_bound 必须在有序范围上使用，否则结果未定义。先 sort 再调用。

⚙️ 3. 变换与累积

transform、accumulate、for_each 是函数式风格编程的核心工具，配合 Lambda 表达式威力倍增。

transform / accumulate / for_each 
#include <algorithm>
#include <numeric>
#include <vector>
#include <string>
#include <iostream>

int main() {
    std::vector<int> v = {1, 2, 3, 4, 5};

    // transform：对每个元素应用函数，结果写入目标范围
    std::vector<int> squares(v.size());
    std::transform(v.begin(), v.end(), squares.begin(),
        [](int x){ return x * x; });
    // squares = {1, 4, 9, 16, 25}

    // transform 双范围版本：对两个序列逐元素操作
    std::vector<int> a = {1, 2, 3}, b = {10, 20, 30};
    std::vector<int> sums(3);
    std::transform(a.begin(), a.end(), b.begin(), sums.begin(),
        [](int x, int y){ return x + y; });
    // sums = {11, 22, 33}

    // accumulate：累积/折叠（在 <numeric> 中）
    int total = std::accumulate(v.begin(), v.end(), 0);
    std::cout << "sum = " << total << "\n";  // 15

    int product = std::accumulate(v.begin(), v.end(), 1,
        [](int acc, int x){ return acc * x; });
    std::cout << "product = " << product << "\n";  // 120

    // 用 accumulate 拼接字符串
    std::vector<std::string> words = {"Hello", " ", "STL", " ", "World"};
    std::string sentence = std::accumulate(words.begin(), words.end(), std::string{});
    std::cout << sentence << "\n";  // "Hello STL World"

    // for_each：对每个元素执行操作（不返回新序列）
    int count_even = 0;
    std::for_each(v.begin(), v.end(), [&count_even](int x){
        if (x % 2 == 0) ++count_even;
    });
    std::cout << "偶数个数: " << count_even << "\n";  // 2

    // C++17 执行策略（并行 for_each）
    // #include <execution>
    // std::for_each(std::execution::par, v.begin(), v.end(), [](int& x){ x *= 2; });

    // count / count_if：统计满足条件的元素数量
    int evens = std::count_if(v.begin(), v.end(), [](int x){ return x % 2 == 0; });
    std::cout << "count_if(偶数) = " << evens << "\n";

    // any_of / all_of / none_of：条件检查
    bool any_neg  = std::any_of(v.begin(), v.end(),  [](int x){ return x < 0; });
    bool all_pos  = std::all_of(v.begin(), v.end(),  [](int x){ return x > 0; });
    bool none_neg = std::none_of(v.begin(), v.end(), [](int x){ return x < 0; });
    std::cout << any_neg << " " << all_pos << " " << none_neg << "\n"; // 0 1 1

    return 0;
}

🗑️ 4. 删除与去重

STL 算法不直接修改容器大小，std::remove 和 std::unique 将"无用"元素移到末尾并返回新的逻辑末端迭代器，需配合容器的 erase 才能真正删除。

remove / remove_if / unique — erase-remove 惯用法 
#include <algorithm>
#include <vector>
#include <iostream>

int main() {
    std::vector<int> v = {1, 2, 3, 2, 4, 2, 5};

    // ❌ 常见误区：remove 不改变容器大小！
    // 它将非目标元素移到前面，返回新的"逻辑末端"迭代器
    auto new_end = std::remove(v.begin(), v.end(), 2);
    // v = {1, 3, 4, 5, ?, ?, ?}  （? 为不确定值）
    // new_end 指向第 5 个位置

    // ✓ erase-remove 惯用法：彻底删除
    v.erase(new_end, v.end());
    // v = {1, 3, 4, 5}

    // 更常见的一行写法
    std::vector<int> u = {1, 2, 3, 2, 4, 2, 5};
    u.erase(std::remove(u.begin(), u.end(), 2), u.end());
    // u = {1, 3, 4, 5}

    // remove_if：按条件删除
    std::vector<int> w = {1, 2, 3, 4, 5, 6, 7, 8};
    w.erase(std::remove_if(w.begin(), w.end(),
        [](int x){ return x % 2 == 0; }),  // 删除所有偶数
        w.end());
    // w = {1, 3, 5, 7}

    // unique：去除连续重复元素（先排序再去重）
    std::vector<int> dup = {3, 1, 4, 1, 5, 9, 2, 6, 5, 3};
    std::sort(dup.begin(), dup.end());
    // dup = {1, 1, 2, 3, 3, 4, 5, 5, 6, 9}
    dup.erase(std::unique(dup.begin(), dup.end()), dup.end());
    // dup = {1, 2, 3, 4, 5, 6, 9}

    for (int x : dup) std::cout << x << " ";
    std::cout << "\n";

    // unique_copy：不修改原序列，去重后输出到新序列
    std::vector<int> src = {1, 1, 2, 3, 3, 3, 4};
    std::vector<int> dst;
    std::unique_copy(src.begin(), src.end(), std::back_inserter(dst));
    // dst = {1, 2, 3, 4}

    return 0;
}
🚨

关键点：std::remove / std::unique 不改变容器的 size()，必须配合 erase 才能真正缩减容器。忘记 erase 是最常见的错误之一。

ℹ️

C++20 新方式：std::erase(container, value) 和 std::erase_if(container, pred) 将 erase-remove 合并为一步，代码更简洁。

🔧 5. 自定义比较器

STL 算法的比较器参数接受任何可调用对象（函数指针、函数对象、Lambda），这让算法能适应复杂的排序和查找逻辑。

Lambda 与函数对象作比较器 
#include <algorithm>
#include <vector>
#include <string>
#include <functional>
#include <iostream>

// 方式一：函数指针
bool by_length(const std::string& a, const std::string& b) {
    return a.size() < b.size();
}

// 方式二：函数对象（仿函数）
struct CaseInsensitiveLess {
    bool operator()(const std::string& a, const std::string& b) const {
        return std::lexicographical_compare(
            a.begin(), a.end(), b.begin(), b.end(),
            [](char c1, char c2){ return std::tolower(c1) < std::tolower(c2); });
    }
};

int main() {
    std::vector<std::string> words = {"banana", "apple", "kiwi", "cherry", "fig"};

    // 方式一：函数指针
    std::sort(words.begin(), words.end(), by_length);
    // {"fig", "kiwi", "apple", "banana", "cherry"}

    // 方式二：函数对象
    std::sort(words.begin(), words.end(), CaseInsensitiveLess{});

    // 方式三：Lambda（最常用，可捕获外部变量）
    std::sort(words.begin(), words.end(),
        [](const std::string& a, const std::string& b){ return a > b; });  // 降序

    // 多键排序：先按长度，再按字典序
    std::sort(words.begin(), words.end(),
        [](const std::string& a, const std::string& b){
            if (a.size() != b.size()) return a.size() < b.size();
            return a < b;
        });

    // std::greater / std::less（标准库函数对象）
    std::sort(words.begin(), words.end(), std::greater<std::string>{});  // 降序

    // 结构体排序示例
    struct Point { int x, y; };
    std::vector<Point> pts = {{3,4},{1,2},{3,1},{2,5}};

    // 先按 x 升序，x 相同按 y 升序
    std::sort(pts.begin(), pts.end(), [](const Point& a, const Point& b){
        return a.x != b.x ? a.x < b.x : a.y < b.y;
    });

    // min_element / max_element 使用自定义比较器
    auto longest = std::max_element(words.begin(), words.end(),
        [](const std::string& a, const std::string& b){ return a.size() < b.size(); });
    std::cout << "最长单词: " << *longest << "\n";

    // nth_element：将第 n 小的元素放到第 n 位（部分排序，O(n)）
    std::vector<int> nums = {5, 2, 8, 1, 9, 3};
    std::nth_element(nums.begin(), nums.begin() + 2, nums.end());
    std::cout << "第 3 小的元素: " << nums[2] << "\n";  // 3

    return 0;
}
💡

严格弱序要求：自定义比较器必须满足严格弱序（Strict Weak Ordering）：不自反（comp(a,a) 为 false）、反对称、传递。违反此规则会导致未定义行为（通常是崩溃）。

综合示例：成绩排行榜 
#include <algorithm>
#include <numeric>
#include <vector>
#include <string>
#include <iostream>
#include <iomanip>

struct Student {
    std::string name;
    int math, english, science;
    int total() const { return math + english + science; }
};

int main() {
    std::vector<Student> students = {
        {"Alice", 95, 88, 92},
        {"Bob",   72, 95, 78},
        {"Cara",  88, 76, 95},
        {"Dan",   95, 88, 80},
    };

    // 按总分降序，总分相同按姓名升序
    std::stable_sort(students.begin(), students.end(),
        [](const Student& a, const Student& b){
            return a.total() != b.total()
                ? a.total() > b.total()
                : a.name < b.name;
        });

    // 用 transform + accumulate 计算各科平均分
    auto avg = [&](auto member_fn) {
        int sum = std::accumulate(students.begin(), students.end(), 0,
            [&](int acc, const Student& s){ return acc + member_fn(s); });
        return static_cast<double>(sum) / students.size();
    };

    std::cout << "排名  姓名    总分\n";
    std::cout << "----  ------  ----\n";
    int rank = 1;
    std::for_each(students.begin(), students.end(), [&rank](const Student& s){
        std::cout << std::setw(4) << rank++ << "  "
                  << std::setw(6) << std::left << s.name
                  << std::setw(4) << std::right << s.total() << "\n";
    });

    // 找出数学最高分的学生
    auto top_math = std::max_element(students.begin(), students.end(),
        [](const Student& a, const Student& b){ return a.math < b.math; });
    std::cout << "\n数学最高分: " << top_math->name << " (" << top_math->math << ")\n";

    // 找出是否有人三科都超过 85 分
    bool all_excellent = std::any_of(students.begin(), students.end(),
        [](const Student& s){ return s.math>85 && s.english>85 && s.science>85; });
    std::cout << "三科全优: " << std::boolalpha << all_excellent << "\n";

    return 0;
}
