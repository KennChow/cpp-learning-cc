📝 1. 文件输出 — ofstream

std::ofstream（output file stream）用于向文件写入数据。它是 std::ostream 的子类，支持所有熟悉的 << 操作符。

写入文本文件 
#include <fstream>
#include <string>
#include <iostream>

int main() {
    // 打开文件（若不存在则创建，若存在则截断）
    std::ofstream ofs("output.txt");

    if (!ofs) {  // 检查是否成功打开
        std::cerr << "无法打开文件！\n";
        return 1;
    }

    // 像 cout 一样写入
    ofs << "Hello, File I/O!\n";
    ofs << "第二行：" << 42 << "\n";

    // 追加模式（不截断原有内容）
    std::ofstream ofs_append("log.txt", std::ios::app);
    ofs_append << "[INFO] 程序已启动\n";

    // 文件会在 ofstream 析构时自动关闭
    return 0;
}
// 也可以手动关闭
// ofs.close();
写入二进制文件 
#include <fstream>
#include <vector>

struct Record {
    int id;
    double value;
    char name[32];
};

int main() {
    // std::ios::binary 以二进制模式打开
    std::ofstream ofs("data.bin", std::ios::binary);

    Record r{1, 3.14, "sensor_A"};
    // write(指针, 字节数)：以原始字节写入
    ofs.write(reinterpret_cast<const char*>(&r), sizeof(r));

    // 写入整数数组
    std::vector<int> nums = {10, 20, 30, 40, 50};
    ofs.write(reinterpret_cast<const char*>(nums.data()),
              nums.size() * sizeof(int));

    return 0;
}
💡

常用打开模式标志：
std::ios::out — 写入（默认）；std::ios::app — 追加；std::ios::binary — 二进制；std::ios::trunc — 截断（默认与 out 同用）。可用 | 组合：std::ios::out | std::ios::binary。

📖 2. 文件输入 — ifstream

std::ifstream（input file stream）用于从文件读取数据，是 std::istream 的子类。

逐行读取文本文件 
#include <fstream>
#include <string>
#include <iostream>
#include <vector>

int main() {
    std::ifstream ifs("output.txt");

    if (!ifs) {
        std::cerr << "文件不存在或无权限读取\n";
        return 1;
    }

    // 方式1：getline 逐行读取（最常用）
    std::string line;
    int lineNo = 0;
    while (std::getline(ifs, line)) {
        ++lineNo;
        std::cout << lineNo << ": " << line << "\n";
    }

    // 方式2：逐词读取（以空白分隔）
    ifs.clear();          // 清除 eof 标志
    ifs.seekg(0);         // 回到文件开头
    std::string word;
    std::vector<std::string> words;
    while (ifs >> word) {
        words.push_back(word);
    }

    // 方式3：一次读入整个文件
    ifs.clear();
    ifs.seekg(0);
    std::string content(
        (std::istreambuf_iterator<char>(ifs)),
        std::istreambuf_iterator<char>()
    );
    std::cout << "文件内容：\n" << content;

    return 0;
}
读取二进制文件 
#include <fstream>

struct Record {
    int id;
    double value;
    char name[32];
};

int main() {
    std::ifstream ifs("data.bin", std::ios::binary);

    Record r;
    // read(指针, 字节数)：从文件读取原始字节
    if (ifs.read(reinterpret_cast<char*>(&r), sizeof(r))) {
        std::cout << "id=" << r.id
                  << " value=" << r.value
                  << " name=" << r.name << "\n";
    }

    // 读取整数数组
    int nums[5];
    ifs.read(reinterpret_cast<char*>(nums), sizeof(nums));
    for (int n : nums) std::cout << n << " ";
    std::cout << "\n";

    return 0;
}

🔍 3. 文件流状态标志

每个流对象内部维护一组状态标志，用于反映当前操作是否成功。正确检查状态是健壮 I/O 代码的基础。

流状态标志与错误检查 
#include <fstream>
#include <iostream>

void printState(const std::ifstream& ifs) {
    std::cout << "good=" << ifs.good()   // 一切正常（所有位均清除）
              << " eof="  << ifs.eof()    // 已到达文件末尾
              << " fail=" << ifs.fail()   // 操作失败（格式错误/eof）
              << " bad="  << ifs.bad()    // 底层 I/O 错误（硬件/OS）
              << "\n";
}

int main() {
    std::ifstream ifs("numbers.txt");  // 文件内容："10 20 abc"

    int n;
    while (ifs >> n) {                 // 读成功时返回 true
        std::cout << "读到: " << n << "\n";
    }
    // 读取 "abc" 失败后：fail=1, eof=0, bad=0

    printState(ifs);

    // 清除错误标志以继续使用流
    ifs.clear();

    // 重新定位到文件开头
    ifs.seekg(0, std::ios::beg);

    // 推荐做法：先检查 !ifs 而非依赖 eof()
    // 常见错误：while (!ifs.eof()) 在最后一行会多循环一次
    // 正确做法：while (std::getline(ifs, line))

    // 获取文件大小
    ifs.seekg(0, std::ios::end);
    std::streamsize size = ifs.tellg();
    std::cout << "文件大小: " << size << " 字节\n";

    return 0;
}
⚠️

常见陷阱：避免用 while (!ifs.eof()) 作为循环条件。eof 标志只有在尝试读取超过文件末尾后才设置，这会导致最后一次读取被处理两次。始终把读取操作本身放在条件中：while (std::getline(ifs, line))。

🚨 4. 异常处理 — try/catch/throw

C++ 异常机制提供了将错误检测与错误处理解耦的方式。throw 抛出异常，try 块标记可能出错的代码，catch 处理异常。

基本 try/catch/throw 
#include <stdexcept>
#include <iostream>
#include <string>

double divide(double a, double b) {
    if (b == 0.0)
        throw std::invalid_argument("除数不能为零");
    return a / b;
}

std::string readLine(const std::string& path) {
    std::ifstream ifs(path);
    if (!ifs)
        throw std::runtime_error("无法打开文件: " + path);
    std::string line;
    std::getline(ifs, line);
    return line;
}

int main() {
    // 单个 catch
    try {
        double result = divide(10.0, 0.0);
        std::cout << result << "\n";
    } catch (const std::invalid_argument& e) {
        std::cerr << "参数错误: " << e.what() << "\n";
    }

    // 多个 catch（从具体到通用排列）
    try {
        std::string line = readLine("nonexistent.txt");
        int n = std::stoi(line);   // 若非数字抛出 invalid_argument
        std::cout << "读到数字: " << n << "\n";
    } catch (const std::runtime_error& e) {
        std::cerr << "运行时错误: " << e.what() << "\n";
    } catch (const std::invalid_argument& e) {
        std::cerr << "格式错误: " << e.what() << "\n";
    } catch (const std::exception& e) {
        // 捕获所有标准异常的兜底处理
        std::cerr << "未知标准异常: " << e.what() << "\n";
    } catch (...) {
        // 捕获任何类型的异常（包括非标准异常）
        std::cerr << "完全未知的异常\n";
    }

    return 0;
}

标准库的异常继承层次：

  • std::exception（根基类，提供 what()）
  • ├── std::logic_error：程序逻辑错误（可预防）
  • │　　├── std::invalid_argument、std::out_of_range、std::length_error
  • └── std::runtime_error：运行时错误（不可预防）
  • 　　　├── std::range_error、std::overflow_error、std::underflow_error
💡

catch 顺序规则：始终先捕获更具体的子类，再捕获基类。若先写 catch (std::exception&)，后续的子类 catch 永远不会被触发。

🏗️ 5. 自定义异常类

从 std::exception 或其子类派生，可以创建携带更丰富上下文信息的自定义异常。

自定义异常类设计 
#include <stdexcept>
#include <string>
#include <iostream>

// 应用级基类：继承自 std::runtime_error
class AppError : public std::runtime_error {
public:
    explicit AppError(const std::string& msg)
        : std::runtime_error("[AppError] " + msg) {}
};

// 文件相关错误：携带路径信息
class FileError : public AppError {
    std::string path_;
public:
    FileError(const std::string& path, const std::string& detail)
        : AppError("文件操作失败 '" + path + "': " + detail)
        , path_(path) {}

    const std::string& path() const noexcept { return path_; }
};

// 解析错误：携带行号信息
class ParseError : public AppError {
    int lineNo_;
public:
    ParseError(int lineNo, const std::string& detail)
        : AppError("第 " + std::to_string(lineNo) + " 行解析失败: " + detail)
        , lineNo_(lineNo) {}

    int lineNo() const noexcept { return lineNo_; }
};

// 使用示例
void processFile(const std::string& path) {
    std::ifstream ifs(path);
    if (!ifs) throw FileError(path, "文件不存在或无权限");

    std::string line;
    int lineNo = 0;
    while (std::getline(ifs, line)) {
        ++lineNo;
        if (line.empty()) throw ParseError(lineNo, "不允许空行");
        // ... 处理 line
    }
}

int main() {
    try {
        processFile("config.txt");
    } catch (const FileError& e) {
        std::cerr << "路径: " << e.path() << "\n";
        std::cerr << "错误: " << e.what() << "\n";
    } catch (const ParseError& e) {
        std::cerr << "行号: " << e.lineNo() << "\n";
        std::cerr << "错误: " << e.what() << "\n";
    } catch (const AppError& e) {
        std::cerr << "应用错误: " << e.what() << "\n";
    }
    return 0;
}
ℹ️

设计建议：自定义异常类应标记析构函数和 what() 为 noexcept（继承自 std::exception 时已自动满足）。不要从 std::exception 直接派生并手动存储字符串——优先继承 std::runtime_error 或 std::logic_error，由基类管理字符串存储。

🛡️ 6. RAII 与异常安全

RAII（Resource Acquisition Is Initialization）：在构造函数中获取资源，在析构函数中释放资源。当异常发生时，栈展开（stack unwinding）会自动调用已构造对象的析构函数，从而保证资源不泄漏。

RAII 文件资源管理 
#include <fstream>
#include <stdexcept>
#include <iostream>

// 不好的做法：手动管理，异常时可能泄漏
void badWrite(const std::string& path) {
    FILE* fp = fopen(path.c_str(), "w");
    // 如果下面抛出异常，fclose 永远不会被调用！
    if (someCondition) throw std::runtime_error("操作失败");
    fprintf(fp, "hello");
    fclose(fp);  // 可能跳过
}

// 好的做法：ofstream 是 RAII 对象
void goodWrite(const std::string& path) {
    std::ofstream ofs(path);          // 构造时打开文件
    if (!ofs) throw std::runtime_error("无法打开: " + path);
    ofs << "hello\n";
    // 即使下面抛出异常，ofs 析构时也会自动关闭文件
    if (someCondition) throw std::runtime_error("操作失败");
    // ofs 在作用域结束时析构，文件被关闭
}

// 异常安全保证等级（从弱到强）
class DataStore {
    std::vector<int> data_;
    std::string filename_;
public:
    DataStore(const std::string& file) : filename_(file) {}

    // 基本保证：异常后对象仍处于有效（但可能变化的）状态
    void appendBasic(int val) {
        data_.push_back(val);      // push_back 可能抛出 bad_alloc
        save();                    // save 可能抛出，此时 data_ 已改变
    }

    // 强保证（commit-or-rollback）：异常后状态完全不变
    void appendStrong(int val) {
        auto backup = data_;       // 先备份（copy-and-swap 惯用法）
        backup.push_back(val);
        saveToTemp();              // 先写临时文件
        std::swap(data_, backup);  // 操作成功后才交换（不抛）
        renameTemp();              // 原子替换（接近不抛）
    }

    // noexcept 保证：承诺绝不抛出
    void clear() noexcept {
        data_.clear();             // vector::clear 不抛
    }

private:
    void save() { /* 写文件 */ }
    void saveToTemp() { /* 写临时文件 */ }
    void renameTemp() { /* 重命名 */ }
    bool someCondition = false;
};
💡

三种异常安全保证：
1. 基本保证：异常后对象仍可析构和赋值，无资源泄漏，但状态可能改变。
2. 强保证（commit-or-rollback）：异常后状态与调用前完全相同。
3. noexcept 保证：该操作绝不抛出异常（移动构造、swap 等应尽量满足）。

综合示例：异常安全的文件处理 
#include <fstream>
#include <stdexcept>
#include <string>
#include <vector>
#include <iostream>

// 异常安全地读取 CSV 中的整数列
std::vector<int> readIntegers(const std::string& path) {
    std::ifstream ifs(path);
    if (!ifs)
        throw std::runtime_error("无法打开文件: " + path);

    std::vector<int> result;
    std::string line;
    int lineNo = 0;

    while (std::getline(ifs, line)) {
        ++lineNo;
        try {
            result.push_back(std::stoi(line));
        } catch (const std::invalid_argument&) {
            throw std::runtime_error(
                "第 " + std::to_string(lineNo) +
                " 行格式错误: '" + line + "' 不是整数");
        } catch (const std::out_of_range&) {
            throw std::runtime_error(
                "第 " + std::to_string(lineNo) +
                " 行数值超出 int 范围");
        }
    }

    if (!ifs.eof())  // 非 eof 结束说明读取中途出错
        throw std::runtime_error("读取文件时发生 I/O 错误");

    return result;  // NRVO，无额外拷贝
}

int main() {
    try {
        auto nums = readIntegers("numbers.csv");
        std::cout << "共读取 " << nums.size() << " 个整数\n";
        for (int n : nums) std::cout << n << " ";
        std::cout << "\n";
    } catch (const std::exception& e) {
        std::cerr << "错误: " << e.what() << "\n";
        return 1;
    }
    return 0;
}
