📦 1. vector — 动态数组

std::vector 是最常用的 STL 容器，本质上是一个连续内存的动态数组。它支持随机访问，在末尾增删元素的摊销复杂度为 O(1)。

vector 基本用法 
#include <vector>
#include <iostream>

int main() {
    // 构造方式
    std::vector<int> v1;               // 空 vector
    std::vector<int> v2(5, 0);         // 5 个元素，全为 0
    std::vector<int> v3 = {1, 2, 3};  // 初始化列表

    // 添加元素
    v1.push_back(10);       // 在末尾追加（拷贝）
    v1.emplace_back(20);    // 在末尾原地构造（推荐，避免临时对象）
    v1.push_back(30);

    // 访问元素
    std::cout << v1[0]      << "\n";  // 不检查越界（危险！）
    std::cout << v1.at(0)   << "\n";  // 越界抛 std::out_of_range（安全）
    std::cout << v1.front() << "\n";  // 第一个元素
    std::cout << v1.back()  << "\n";  // 最后一个元素

    // 遍历
    for (int x : v1) std::cout << x << " ";  // 范围 for
    std::cout << "\n";

    // 删除
    v1.pop_back();          // 删除末尾元素，O(1)
    v1.erase(v1.begin());   // 删除首元素，O(n)（后续元素前移）

    // 大小
    std::cout << "size: " << v1.size()  << "\n";  // 实际元素数
    std::cout << "empty: " << v1.empty() << "\n"; // 是否为空
    v1.clear();                                      // 清空所有元素

    return 0;
}

⚡ capacity vs size — 扩容机制

size() 是当前实际元素个数；capacity() 是已分配内存可容纳的元素上限。当 push_back 触发扩容时，vector 通常以 2 倍（或 1.5 倍）增长，并将所有元素移动到新内存，这是一次 O(n) 操作。

capacity 与性能优化 
#include <vector>
#include <iostream>

int main() {
    std::vector<int> v;

    // 演示扩容过程
    for (int i = 0; i < 10; ++i) {
        v.push_back(i);
        std::cout << "size=" << v.size()
                  << " capacity=" << v.capacity() << "\n";
    }
    // 典型输出：capacity 在 1,2,4,8... 时翻倍

    // 预分配内存，避免多次扩容（O(n) 操作次数从多次降为 1 次）
    std::vector<int> v2;
    v2.reserve(1000);  // 预留 1000 个元素的空间，size 不变
    std::cout << "size=" << v2.size() << " capacity=" << v2.capacity() << "\n";

    // 插入 1000 个元素，不触发任何扩容
    for (int i = 0; i < 1000; ++i) v2.push_back(i);

    // 释放多余容量（C++11）
    v2.shrink_to_fit();  // capacity 收缩至 size

    // resize vs reserve
    std::vector<int> v3;
    v3.resize(5);   // 改变 size 为 5，新元素值初始化为 0
    v3.reserve(10); // 只改变 capacity，size 不变

    return 0;
}
⚠️

迭代器失效：对 vector 调用 push_back、insert、erase 等操作后，若发生扩容，所有指向该 vector 元素的迭代器、指针和引用都将失效。在循环中修改 vector 要格外小心。

🔧 push_back vs emplace_back

emplace_back 原地构造 
#include <vector>
#include <string>

struct Point {
    int x, y;
    Point(int x, int y) : x(x), y(y) {}
};

int main() {
    std::vector<Point> pts;

    // push_back：先构造临时对象，再移动/拷贝进容器
    pts.push_back(Point(1, 2));  // 构造临时 Point，再移动

    // emplace_back：直接在容器内存中原地构造，无临时对象
    pts.emplace_back(3, 4);  // 直接传参数，更高效

    // 对于字符串同理
    std::vector<std::string> words;
    words.push_back("hello");          // 从字面量构造 string 临时对象，再拷贝
    words.emplace_back("world");       // 直接在容器中构造 string，推荐
    words.emplace_back(5, 'a');        // 构造 "aaaaa"，只有 emplace_back 能这样用

    return 0;
}

🔗 2. list / deque — 链表与双端队列

当需要在序列中间频繁插入或删除元素时，std::list 是更好的选择；当需要在两端高效增删时，std::deque 比 vector 更合适。

list — 双向链表 
#include <list>
#include <iostream>
#include <algorithm>

int main() {
    std::list<int> lst = {3, 1, 4, 1, 5, 9, 2, 6};

    // 两端插入/删除，O(1)
    lst.push_front(0);   // 头部插入
    lst.push_back(7);    // 尾部插入
    lst.pop_front();     // 删除头部
    lst.pop_back();      // 删除尾部

    // 任意位置插入：先找到迭代器，再插入（插入本身 O(1)，但查找 O(n)）
    auto it = std::find(lst.begin(), lst.end(), 5);
    if (it != lst.end()) {
        lst.insert(it, 99);   // 在 5 之前插入 99
        lst.erase(it);        // 删除 5（迭代器有效！）
    }

    // list 特有操作
    lst.sort();              // 排序（不能用 std::sort，因为不支持随机访问）
    lst.reverse();           // 反转
    lst.unique();            // 删除连续重复元素（排序后才有意义）
    lst.remove(1);           // 删除所有值为 1 的元素

    // 合并两个有序链表
    std::list<int> lst2 = {2, 4, 6};
    lst.merge(lst2);         // lst 必须已排序，合并后 lst2 变空

    for (int x : lst) std::cout << x << " ";
    std::cout << "\n";

    // 注意：list 不支持随机访问（没有 operator[]）
    // lst[0] 是错误的！

    return 0;
}
deque — 双端队列 
#include <deque>
#include <iostream>

int main() {
    std::deque<int> dq = {3, 4, 5};

    // 两端插入/删除，O(1)（vector 只有末尾 O(1)）
    dq.push_front(2);  // 头部插入
    dq.push_front(1);
    dq.push_back(6);   // 尾部插入
    dq.pop_front();    // 删除头部

    // 支持随机访问（与 vector 相同）
    std::cout << dq[0] << "\n";   // OK
    std::cout << dq.at(1) << "\n"; // OK，越界检查

    for (int x : dq) std::cout << x << " ";
    std::cout << "\n";

    // deque 的底层是分段连续内存，不像 vector 是单块连续内存
    // 因此 push_front 是 O(1)，但随机访问稍慢于 vector

    // 适合实现：滑动窗口、BFS 队列、双端任务队列

    return 0;
}
内存结构对比
vector： 1 2 3 4 空 空 ← 单块连续内存
list   ： [prev|1|next] ↔ [prev|2|next] ↔ [prev|3|next] ← 分散节点，双向链接
deque ： [...|1|2] → [3|4|5] → [6|...] ← 分段连续内存块

🗺️ 3. map / unordered_map — 有序 / 无序映射

std::map 基于红黑树实现，键按升序自动排列，查找复杂度 O(log n)。std::unordered_map 基于哈希表，查找平均 O(1)，但键无序且最坏 O(n)。

map 基本操作 
#include <map>
#include <iostream>
#include <string>

int main() {
    std::map<std::string, int> scores;

    // 插入方式
    scores["Alice"] = 95;              // operator[]：键不存在则插入默认值（0），再赋值
    scores.insert({"Bob", 87});        // insert：键已存在时不更新
    scores.emplace("Charlie", 92);     // emplace：原地构造，推荐

    // 访问
    std::cout << scores["Alice"] << "\n";  // 95
    // 注意：operator[] 会在键不存在时插入 0，可能造成意外！

    // 安全查找（不插入）
    auto it = scores.find("Dave");
    if (it != scores.end()) {
        std::cout << it->second << "\n";
    } else {
        std::cout << "Dave not found\n";
    }

    // C++20：contains()
    if (scores.contains("Alice")) {
        std::cout << "Alice exists\n";
    }

    // 遍历（按键升序）
    for (const auto& [name, score] : scores) {  // 结构化绑定 (C++17)
        std::cout << name << ": " << score << "\n";
    }

    // 删除
    scores.erase("Bob");              // 按键删除
    scores.erase(scores.find("Alice")); // 按迭代器删除

    std::cout << "size: " << scores.size() << "\n";

    return 0;
}
unordered_map — 哈希映射 
#include <unordered_map>
#include <iostream>
#include <string>

int main() {
    // 接口与 map 几乎相同，但内部用哈希表
    std::unordered_map<std::string, int> word_count;

    // 统计词频（经典用法）
    std::vector<std::string> words = {"apple", "banana", "apple", "cherry", "banana", "apple"};
    for (const auto& w : words) {
        ++word_count[w];  // 键不存在时自动插入 0，然后 ++
    }

    for (const auto& [word, cnt] : word_count) {
        std::cout << word << ": " << cnt << "\n";
    }
    // 注意：输出顺序是哈希顺序，不保证有序

    // 性能相关：预分配桶数，避免频繁 rehash
    word_count.reserve(1000);          // 预留桶，减少 rehash
    std::cout << "bucket_count: " << word_count.bucket_count() << "\n";
    std::cout << "load_factor: "  << word_count.load_factor()  << "\n";

    // 自定义哈希（用于自定义类型作 key）
    struct PairHash {
        size_t operator()(const std::pair<int,int>& p) const {
            return std::hash<int>{}(p.first) ^ (std::hash<int>{}(p.second) << 16);
        }
    };
    std::unordered_map<std::pair<int,int>, std::string, PairHash> grid;
    grid[{0, 0}] = "origin";
    grid[{1, 2}] = "point";

    return 0;
}
💡

map vs unordered_map 选择原则：若需要有序遍历或范围查询（如找所有 key 在 [a, b] 之间的条目），用 map。若只需点查且追求最快速度，用 unordered_map（平均 O(1)）。自定义类型作 key 时，map 只需 operator<，unordered_map 还需提供哈希函数。

🔵 4. set / unordered_set — 集合

集合存储不重复的元素。std::set 基于红黑树，元素自动有序；std::unordered_set 基于哈希表，无序但查找更快。

set 与 unordered_set 用法 
#include <set>
#include <unordered_set>
#include <iostream>
#include <vector>

int main() {
    // set：有序，自动去重，O(log n) 操作
    std::set<int> s = {5, 3, 1, 4, 2, 3, 5};  // 重复元素被自动丢弃
    // s 内容：{1, 2, 3, 4, 5}（升序）

    s.insert(6);          // O(log n)
    s.erase(3);           // O(log n)
    bool found = s.count(4);  // 0 或 1（因为不重复）

    // 有序集合的范围操作
    auto lo = s.lower_bound(2);  // 第一个 >= 2 的迭代器
    auto hi = s.upper_bound(4);  // 第一个 >  4 的迭代器
    for (auto it = lo; it != hi; ++it)
        std::cout << *it << " ";  // 输出 2 3 4
    std::cout << "\n";

    for (int x : s) std::cout << x << " ";  // 按升序遍历
    std::cout << "\n";

    // unordered_set：无序，O(1) 平均查找
    std::unordered_set<std::string> visited;
    visited.insert("A");
    visited.insert("B");
    visited.insert("A");  // 重复，插入被忽略
    std::cout << visited.size() << "\n";  // 2

    if (visited.find("A") != visited.end()) {
        std::cout << "A already visited\n";
    }

    // 经典用法：用 vector 去重
    std::vector<int> v = {3, 1, 4, 1, 5, 9, 2, 6, 5};
    std::unordered_set<int> seen(v.begin(), v.end());
    std::vector<int> unique_v(seen.begin(), seen.end());
    // unique_v 中无重复元素（但顺序不定）

    return 0;
}
📝

multiset / multimap：若需要允许重复键，使用 std::multiset 和 std::multimap。它们与 set/map 接口类似，但 insert 总会成功，count 可返回大于 1 的值。equal_range(key) 可获取所有重复键的范围。

🧭 5. 容器选择指南

选择容器的核心依据：主要操作是什么（随机访问、顺序遍历、中间插删、键值查找、去重）以及对顺序是否有要求。

容器 随机访问 头部插删 尾部插删 中间插删 查找 有序
vector O(1) O(n) O(1)* O(n) O(n) 插入顺序
deque O(1) O(1) O(1) O(n) O(n) 插入顺序
list O(n) O(1) O(1) O(1)** O(n) 插入顺序
map — — — O(log n) O(log n) 键升序
unordered_map — — — O(1)* O(1)* 无序
set — — — O(log n) O(log n) 元素升序
unordered_set — — — O(1)* O(1)* 无序

* 摊销 O(1)（偶尔触发扩容/rehash 为 O(n)）；** 需先找到迭代器位置，查找本身 O(n)

决策示例 
// 场景 1：需要随机访问，主要在末尾增删
// → 用 vector
std::vector<int> data;

// 场景 2：需要在两端频繁增删（如滑动窗口、BFS）
// → 用 deque
std::deque<int> window;

// 场景 3：频繁在中间插入/删除，不需要随机访问
// → 用 list
std::list<Task> task_queue;

// 场景 4：需要键值映射，且需要按键有序遍历或范围查询
// → 用 map
std::map<std::string, int> sorted_scores;

// 场景 5：需要键值映射，只需快速点查，不关心顺序
// → 用 unordered_map（O(1) vs O(log n)，数据量大时差异显著）
std::unordered_map<std::string, int> fast_lookup;

// 场景 6：需要去重，并需要按序遍历或范围查询
// → 用 set
std::set<int> ordered_unique;

// 场景 7：需要快速判断元素是否存在（去重/visited 标记）
// → 用 unordered_set
std::unordered_set<int> visited;

// 经验法则：
// - 默认先用 vector，不满足再换
// - 需要键值对？用 unordered_map（默认），需排序再换 map
// - 需要去重？用 unordered_set（默认），需排序再换 set
💡

实测为王：理论复杂度不等于实际性能。vector 因缓存友好（连续内存），实际上在小数据量下常常比 list 更快，即使理论上 list 的中间插入是 O(1)。遇到性能瓶颈时，先 profile 再换容器。

🧪 综合代码示例：学生成绩管理

student_scores.cpp 
#include <iostream>
#include <vector>
#include <map>
#include <unordered_map>
#include <set>
#include <string>
#include <algorithm>

struct Student {
    std::string name;
    int score;
};

int main() {
    // 原始数据：用 vector 存储，保持插入顺序
    std::vector<Student> students = {
        {"Alice", 95}, {"Bob", 72}, {"Charlie", 88},
        {"Dave", 72},  {"Eve", 95}, {"Frank", 60}
    };

    // 需求 1：按分数从高到低排序（用 sort + lambda）
    std::sort(students.begin(), students.end(),
              [](const Student& a, const Student& b){ return a.score > b.score; });
    std::cout << "=== 排名 ===\n";
    for (const auto& s : students)
        std::cout << s.name << ": " << s.score << "\n";

    // 需求 2：快速按姓名查分（用 unordered_map，O(1) 查找）
    std::unordered_map<std::string, int> name_to_score;
    for (const auto& s : students)
        name_to_score[s.name] = s.score;

    std::cout << "\n=== 查分 ===\n";
    std::cout << "Charlie: " << name_to_score["Charlie"] << "\n";

    // 需求 3：统计各分数段人数（用 map，有序）
    std::map<std::string, int> grade_count;
    for (const auto& s : students) {
        if      (s.score >= 90) grade_count["A"]++;
        else if (s.score >= 80) grade_count["B"]++;
        else if (s.score >= 70) grade_count["C"]++;
        else                    grade_count["D"]++;
    }
    std::cout << "\n=== 分数段分布 ===\n";
    for (const auto& [grade, cnt] : grade_count)
        std::cout << grade << ": " << cnt << " 人\n";

    // 需求 4：有哪些不重复的分数（用 set，自动去重且有序）
    std::set<int> unique_scores;
    for (const auto& s : students)
        unique_scores.insert(s.score);

    std::cout << "\n=== 出现的分数 ===\n";
    for (int score : unique_scores)
        std::cout << score << " ";
    std::cout << "\n";

    return 0;
}
