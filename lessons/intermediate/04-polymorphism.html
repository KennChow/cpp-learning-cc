🔀 什么是多态？

多态（Polymorphism）的字面含义是"多种形态"。在 C++ 中，多态允许我们通过基类的指针或引用，在运行时自动调用正确的派生类方法——即使我们在编写代码时不知道具体是哪个派生类。

多态的威力：同一接口，不同实现 
#include <iostream>
#include <vector>
using namespace std;

class Animal {
public:
    virtual void speak() { cout << "...\n"; }
    virtual ~Animal() {}  // 虚析构函数（重要！）
};

class Dog : public Animal {
public:
    void speak() override { cout << "汪汪！\n"; }
};

class Cat : public Animal {
public:
    void speak() override { cout << "喵喵！\n"; }
};

class Duck : public Animal {
public:
    void speak() override { cout << "嘎嘎！\n"; }
};

// 多态的核心：通过基类指针调用
void makeNoise(Animal* animal) {
    animal->speak();  // 运行时决定调用哪个版本！
}

int main() {
    // 不同类型的对象，通过统一接口操作
    vector<Animal*> zoo = {
        new Dog(), new Cat(), new Duck(), new Dog()
    };

    for (Animal* a : zoo) {
        makeNoise(a);    // 多态！自动调用正确的 speak()
    }
    // 输出：汪汪！ 喵喵！ 嘎嘎！ 汪汪！

    for (Animal* a : zoo) delete a;
    return 0;
}

多态的三个必要条件：

  • 必须有继承关系
  • 父类中声明 virtual 函数
  • 通过基类指针或引用调用虚函数（值类型不行）

⚠️ 没有 virtual 会怎样？

如果不加 virtual，通过基类指针调用的函数将在编译时根据指针类型决定，而非运行时根据对象类型决定——即静态绑定。

静态绑定 vs 动态绑定对比 
#include <iostream>
using namespace std;

class Base {
public:
    void nonVirtual() { cout << "Base::nonVirtual\n"; }
    virtual void isVirtual() { cout << "Base::isVirtual\n"; }
};

class Derived : public Base {
public:
    void nonVirtual() { cout << "Derived::nonVirtual\n"; }
    void isVirtual() override { cout << "Derived::isVirtual\n"; }
};

int main() {
    Derived d;
    Base* ptr = &d;  // 基类指针，指向派生类对象

    ptr->nonVirtual();  // 静态绑定：调用 Base::nonVirtual（看指针类型）
    ptr->isVirtual();   // 动态绑定：调用 Derived::isVirtual（看对象类型）

    // 输出：
    // Base::nonVirtual     ← 没有 virtual，调用了"错误"的版本
    // Derived::isVirtual   ← 有 virtual，正确的多态行为

    // 通过引用同样触发多态
    Base& ref = d;
    ref.isVirtual();    // Derived::isVirtual（动态绑定）

    // 通过值则不触发多态（对象切片！）
    Base val = d;       // 对象切片：丢失了 Derived 的部分
    val.isVirtual();    // Base::isVirtual（静态绑定）

    return 0;
}
🚫

对象切片（Object Slicing）：将派生类对象赋值给基类值变量时，派生类特有的数据成员会被"切掉"，只保留基类部分。这是一个常见的 bug 来源。多态必须通过指针或引用，永远不要按值传递多态对象。

📊 虚函数表（vtable）工作原理

虚函数的动态分发是通过虚函数表（vtable，Virtual Table）实现的。这是 C++ 运行时多态的底层机制：

  • 每个包含虚函数的类都有一张 vtable（函数指针数组）
  • 每个对象包含一个隐藏的 vptr（虚指针），指向所属类的 vtable
  • 调用虚函数时：通过 vptr 找到 vtable，再通过 vtable 找到函数指针，最后调用

vtable 动态分发：Dog 对象的 vptr 指向 Dog 的 vtable，因此调用 Dog::speak()

vtable 的开销 
// vtable 的内存开销（一般每个对象多一个指针）
#include <iostream>
using namespace std;

class WithoutVirtual {
    int x, y;
};

class WithVirtual {
    int x, y;
    virtual void foo() {}
};

int main() {
    cout << "无虚函数对象大小: " << sizeof(WithoutVirtual) << " 字节\n";  // 8
    cout << "有虚函数对象大小: " << sizeof(WithVirtual)    << " 字节\n";  // 16（多了vptr）
    // 在 64 位系统上，vptr 占 8 字节
    return 0;
}

📐 纯虚函数与抽象类

纯虚函数（= 0）定义了一个接口规范，派生类必须提供实现。包含纯虚函数的类称为抽象类，不能直接实例化。

Shape 抽象基类完整示例 
#include <iostream>
#include <cmath>
#include <vector>
using namespace std;

// 抽象基类：定义接口规范
class Shape {
public:
    virtual double area()      = 0;  // 纯虚函数：必须由子类实现
    virtual double perimeter() = 0;  // 纯虚函数

    // 普通虚函数：有默认实现，子类可选择覆盖
    virtual void print() {
        cout << typeid(*this).name()
             << " | 面积: " << area()
             << " | 周长: " << perimeter() << "\n";
    }

    // 虚析构函数：抽象基类必须有！
    virtual ~Shape() {}
};

// Shape 不能实例化
// Shape s;  // 错误！抽象类不能实例化

class Circle : public Shape {
    double radius;
public:
    Circle(double r) : radius(r) {}
    double area()      override { return M_PI * radius * radius; }
    double perimeter() override { return 2 * M_PI * radius; }
};

class Rectangle : public Shape {
    double w, h;
public:
    Rectangle(double w, double h) : w(w), h(h) {}
    double area()      override { return w * h; }
    double perimeter() override { return 2 * (w + h); }
};

class Triangle : public Shape {
    double a, b, c;  // 三条边
public:
    Triangle(double a, double b, double c) : a(a), b(b), c(c) {}
    double area() override {
        double s = (a + b + c) / 2;
        return sqrt(s * (s-a) * (s-b) * (s-c));  // 海伦公式
    }
    double perimeter() override { return a + b + c; }
};

int main() {
    // 多态：通过基类指针操作不同类型
    vector<Shape*> shapes = {
        new Circle(5),
        new Rectangle(4, 6),
        new Triangle(3, 4, 5)
    };

    double totalArea = 0;
    for (Shape* s : shapes) {
        s->print();      // 多态调用
        totalArea += s->area();
    }
    cout << "总面积: " << totalArea << "\n";

    for (Shape* s : shapes) delete s;
    return 0;
}

🗑️ 虚析构函数的重要性

当通过基类指针 delete 派生类对象时，如果基类析构函数不是虚的，会导致未定义行为（通常是资源泄漏）。

虚析构函数的必要性 
#include <iostream>
using namespace std;

// 危险示例：非虚析构函数
class BadBase {
public:
    ~BadBase() { cout << "BadBase 析构\n"; }
};
class BadDerived : public BadBase {
    int* data;
public:
    BadDerived() : data(new int[100]) { cout << "BadDerived 构造\n"; }
    ~BadDerived() {
        delete[] data;  // 这行永远不会被调用！
        cout << "BadDerived 析构\n";
    }
};

// 安全示例：虚析构函数
class GoodBase {
public:
    virtual ~GoodBase() { cout << "GoodBase 析构\n"; }
};
class GoodDerived : public GoodBase {
    int* data;
public:
    GoodDerived() : data(new int[100]) { cout << "GoodDerived 构造\n"; }
    ~GoodDerived() override {
        delete[] data;  // 正确释放！
        cout << "GoodDerived 析构\n";
    }
};

int main() {
    // 危险：内存泄漏！BadDerived::~BadDerived 不会被调用
    BadBase* bad = new BadDerived();
    delete bad;   // 只调用 BadBase::~BadBase，data 泄漏！

    cout << "---\n";

    // 安全：虚析构确保正确调用链
    GoodBase* good = new GoodDerived();
    delete good;  // 先 GoodDerived::~GoodDerived，再 GoodBase::~GoodBase
    return 0;
}
🚫

黄金法则：如果一个类有任何虚函数（即可能被多态使用），其析构函数必须是虚的。或者更简单地记：设计为基类的类，析构函数必须是 virtual。这条规则几乎没有例外。

🔍 dynamic_cast 与类型安全转型

有时候我们需要将基类指针转换回派生类指针（向下转型）。dynamic_cast 提供了类型安全的转型，失败时返回 nullptr（对指针）或抛出异常（对引用）。

dynamic_cast 安全向下转型 
#include <iostream>
#include <typeinfo>
using namespace std;

class Animal {
public:
    virtual ~Animal() {}
    virtual void speak() = 0;
};

class Dog : public Animal {
public:
    void speak() override { cout << "汪！\n"; }
    void fetch() { cout << "取球！\n"; }
};

class Cat : public Animal {
public:
    void speak() override { cout << "喵！\n"; }
    void purr() { cout << "呼噜...\n"; }
};

void processAnimal(Animal* a) {
    a->speak();  // 多态调用

    // 尝试向下转型为 Dog
    Dog* dog = dynamic_cast<Dog*>(a);
    if (dog) {
        dog->fetch();  // 只有 Dog 才能做的事
    }

    // 尝试向下转型为 Cat
    Cat* cat = dynamic_cast<Cat*>(a);
    if (cat) {
        cat->purr();  // 只有 Cat 才能做的事
    }

    // typeid：获取运行时类型信息
    cout << "实际类型: " << typeid(*a).name() << "\n";
}

int main() {
    Animal* animals[] = { new Dog(), new Cat() };
    for (Animal* a : animals) {
        processAnimal(a);
        cout << "---\n";
        delete a;
    }
    return 0;
}
💡

何时用 dynamic_cast？：在设计良好的多态代码中，很少需要 dynamic_cast——应该通过虚函数让派生类自己处理差异。频繁使用 dynamic_cast 往往意味着设计存在问题（违反开闭原则）。但在处理遗留代码或某些特殊场景下，它是必要的安全工具。

✅ 虚函数接口设计最佳实践

设计多态接口时，遵循以下原则可以让代码更健壮、更易于扩展：

  • 基类析构函数必须是 virtual（已强调）
  • 纯虚接口类：只包含纯虚函数的类相当于 Java/C# 的接口（interface），鼓励使用
  • override 关键字：覆盖虚函数时始终加 override
  • 里氏替换原则（LSP）：子类必须能替换父类，不能破坏父类的行为约定
纯虚接口类的设计 
// 纯虚接口：只有纯虚函数，定义契约
class Drawable {
public:
    virtual void draw() = 0;           // 必须实现：绘制
    virtual void resize(double factor) = 0;  // 必须实现：缩放
    virtual ~Drawable() {}
};

class Serializable {
public:
    virtual string serialize() const = 0;  // 序列化为字符串
    virtual void deserialize(const string& data) = 0;
    virtual ~Serializable() {}
};

// 一个类可以实现多个接口（多重继承的正确用法）
class Icon : public Drawable, public Serializable {
    int x, y, size;
    string label;
public:
    Icon(int x, int y, int sz, string lbl)
        : x(x), y(y), size(sz), label(lbl) {}

    void draw() override {
        cout << "绘制图标 [" << label << "] 位置(" << x << "," << y << ")\n";
    }
    void resize(double factor) override { size *= factor; }
    string serialize() const override {
        return label + ":" + to_string(x) + "," + to_string(y);
    }
    void deserialize(const string& data) override { /* ... */ }
};

理解题

解释"静态绑定"与"动态绑定"的区别，并说明为什么多态只能通过指针或引用触发，而不能通过值类型触发（提示：思考对象切片问题）。

编程题

挑战题
