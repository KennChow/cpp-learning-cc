🔒 1. 单例模式 — Singleton

单例模式保证一个类只有一个实例，并提供全局访问点。现代 C++ 中最简洁且线程安全的实现是 Meyers' Singleton，利用局部静态变量的线程安全初始化（C++11 保证）。

singleton.cpp — Meyers' Singleton（线程安全） 
#include <iostream>
#include <string>
#include <mutex>

// Meyers' Singleton：最简洁的线程安全实现
// C++11 保证局部静态变量只初始化一次，且是线程安全的
class Logger {
public:
    // 获取唯一实例
    static Logger& instance() {
        static Logger logger;   // 首次调用时初始化，线程安全
        return logger;
    }

    void log(const std::string& msg) {
        std::lock_guard<std::mutex> lock(mutex_);
        std::cout << "[LOG] " << msg << "\n";
    }

    void setLevel(int level) {
        std::lock_guard<std::mutex> lock(mutex_);
        level_ = level;
    }

    int level() const { return level_; }

    // 禁止拷贝和赋值
    Logger(const Logger&) = delete;
    Logger& operator=(const Logger&) = delete;

private:
    Logger() : level_(0) {}   // 私有构造函数
    ~Logger() = default;
    std::mutex mutex_;
    int level_;
};

// 使用
int main() {
    Logger::instance().setLevel(2);
    Logger::instance().log("程序启动");
    Logger::instance().log("加载配置完成");

    // 验证是同一个实例
    Logger& a = Logger::instance();
    Logger& b = Logger::instance();
    std::cout << "同一实例？" << (&a == &b ? "是" : "否") << "\n";  // 是
    return 0;
}

// ---- 扩展：模板化单例基类（CRTP 复用） ----
template<typename Derived>
class Singleton {
public:
    static Derived& instance() {
        static Derived inst;
        return inst;
    }
    Singleton(const Singleton&) = delete;
    Singleton& operator=(const Singleton&) = delete;
protected:
    Singleton() = default;
};

class AppConfig : public Singleton<AppConfig> {
    friend class Singleton<AppConfig>;
public:
    std::string dbUrl = "localhost:5432";
    int maxConnections = 10;
private:
    AppConfig() = default;
};

// AppConfig::instance().dbUrl = "prod-db:5432";
注意：单例模式会引入全局状态，增加代码耦合度，难以进行单元测试。在设计时需权衡，仅对真正的全局唯一资源（日志、配置、线程池）使用。

🏭 2. 工厂模式 — Factory Method & Abstract Factory

工厂方法：将对象的创建延迟到子类，调用者无需知道具体类型。抽象工厂：提供创建一族相关对象的接口，保证对象间的兼容性。

factory.cpp — 工厂方法模式 
#include <iostream>
#include <memory>
#include <string>
#include <unordered_map>
#include <functional>

// 产品接口
struct Shape {
    virtual ~Shape() = default;
    virtual double area() const = 0;
    virtual std::string name() const = 0;
};

// 具体产品
struct Circle : Shape {
    explicit Circle(double r) : radius(r) {}
    double area() const override { return 3.14159 * radius * radius; }
    std::string name() const override { return "Circle(r=" + std::to_string(radius) + ")"; }
    double radius;
};

struct Rectangle : Shape {
    Rectangle(double w, double h) : width(w), height(h) {}
    double area() const override { return width * height; }
    std::string name() const override { return "Rectangle(" + std::to_string(width) + "x" + std::to_string(height) + ")"; }
    double width, height;
};

// 方法1：简单工厂函数（现代 C++ 风格）
std::unique_ptr<Shape> createCircle(double r) {
    return std::make_unique<Circle>(r);
}
std::unique_ptr<Shape> createRectangle(double w, double h) {
    return std::make_unique<Rectangle>(w, h);
}

// 方法2：注册式工厂（可扩展，无需修改工厂代码）
class ShapeFactory {
public:
    using Creator = std::function<std::unique_ptr<Shape>()>;

    static ShapeFactory& instance() {
        static ShapeFactory inst;
        return inst;
    }

    void registerShape(const std::string& type, Creator creator) {
        registry_[type] = std::move(creator);
    }

    std::unique_ptr<Shape> create(const std::string& type) const {
        auto it = registry_.find(type);
        if (it == registry_.end())
            throw std::runtime_error("Unknown shape: " + type);
        return it->second();
    }

private:
    std::unordered_map<std::string, Creator> registry_;
};

int main() {
    auto& factory = ShapeFactory::instance();
    factory.registerShape("circle",    [] { return std::make_unique<Circle>(5.0); });
    factory.registerShape("rectangle", [] { return std::make_unique<Rectangle>(3.0, 4.0); });

    for (const std::string& type : {"circle", "rectangle"}) {
        auto shape = factory.create(type);
        std::cout << shape->name() << " 面积=" << shape->area() << "\n";
    }
    return 0;
}

// 抽象产品族
struct Button {
    virtual ~Button() = default;
    virtual void render() const = 0;
};
struct Checkbox {
    virtual ~Checkbox() = default;
    virtual void render() const = 0;
};

// 具体产品：深色主题
struct DarkButton : Button {
    void render() const override { std::cout << "[Dark Button]\n"; }
};
struct DarkCheckbox : Checkbox {
    void render() const override { std::cout << "[Dark Checkbox]\n"; }
};

// 具体产品：浅色主题
struct LightButton : Button {
    void render() const override { std::cout << "[Light Button]\n"; }
};
struct LightCheckbox : Checkbox {
    void render() const override { std::cout << "[Light Checkbox]\n"; }
};

// 抽象工厂接口
struct UIFactory {
    virtual ~UIFactory() = default;
    virtual std::unique_ptr<Button>   createButton()   const = 0;
    virtual std::unique_ptr<Checkbox> createCheckbox() const = 0;
};

// 具体工厂
struct DarkThemeFactory : UIFactory {
    std::unique_ptr<Button>   createButton()   const override { return std::make_unique<DarkButton>(); }

    std::unique_ptr<Button>   createButton()   const override { return std::make_unique<LightButton>(); }
    std::unique_ptr<Checkbox> createCheckbox() const override { return std::make_unique<LightCheckbox>(); }
};

// 客户端代码：只依赖抽象接口
void renderUI(const UIFactory& factory) {
    auto btn = factory.createButton();
    auto chk = factory.createCheckbox();
    btn->render();
    chk->render();
}

👁️ 3. 观察者模式 — Observer

观察者模式定义对象间的一对多依赖：当主题（Subject）状态变化时，所有观察者（Observer）自动收到通知。现代 C++ 用 std::function 实现更灵活的事件系统，无需继承体系。

observer.cpp — 基于 std::function 的事件系统 
#include <iostream>
#include <functional>
#include <vector>
#include <string>
#include <algorithm>

// 通用事件总线：支持任意回调签名
template<typename... Args>
class Event {
public:
    using Handler = std::function<void(Args...)>;
    using HandlerId = size_t;

    // 订阅：返回订阅 ID，供取消订阅使用
    HandlerId subscribe(Handler handler) {
        handlers_.push_back({++nextId_, std::move(handler)});
        return nextId_;
    }

    // 取消订阅
    void unsubscribe(HandlerId id) {
        handlers_.erase(
            std::remove_if(handlers_.begin(), handlers_.end(),
                [id](const auto& h) { return h.id == id; }),
            handlers_.end()
        );
    }

    // 触发事件
    void emit(Args... args) const {
        for (const auto& h : handlers_)
            h.handler(args...);
    }

private:
    struct HandlerEntry {
        HandlerId id;
        Handler handler;
    };
    std::vector<HandlerEntry> handlers_;
    HandlerId nextId_ = 0;
};

// 具体主题：股票价格
class Stock {
public:
    explicit Stock(std::string ticker) : ticker_(std::move(ticker)), price_(0.0) {}

    // 事件：价格变化时触发，传递 (ticker, newPrice, delta)
    Event<std::string, double, double> onPriceChanged;

    void setPrice(double newPrice) {
        double delta = newPrice - price_;
        price_ = newPrice;
        onPriceChanged.emit(ticker_, price_, delta);
    }

    double price() const { return price_; }
    const std::string& ticker() const { return ticker_; }

private:
    std::string ticker_;
    double price_;
};

int main() {
    Stock apple("AAPL");

    // 订阅1：价格显示
    auto id1 = apple.onPriceChanged.subscribe(
        [](const std::string& t, double p, double d) {
            std::cout << t << " 新价格: $" << p
                      << " (变化: " << (d >= 0 ? "+" : "") << d << ")\n";
        });

    // 订阅2：涨跌报警
    auto id2 = apple.onPriceChanged.subscribe(
        [](const std::string& t, double p, double d) {
            if (d > 5.0)  std::cout << "[警报] " << t << " 大幅上涨！\n";
            if (d < -5.0) std::cout << "[警报] " << t << " 大幅下跌！\n";
        });

    apple.setPrice(150.0);   // 触发两个观察者
    apple.setPrice(158.0);   // 触发涨幅报警
    apple.setPrice(152.0);   // 正常变化

    // 取消订阅报警
    apple.onPriceChanged.unsubscribe(id2);
    apple.setPrice(145.0);   // 只有价格显示触发

    return 0;
}

⚔️ 4. 策略模式 — Strategy

策略模式将算法封装为独立对象，使其可以互换。C++ 中有两种实现：运行时策略（虚函数 / std::function，灵活）和编译期策略（模板参数，零开销）。

strategy.cpp — 运行时策略（std::function） 
#include <iostream>
#include <functional>
#include <algorithm>
#include <vector>
#include <string>
#include <numeric>

// 运行时策略：用 std::function 持有算法
class DataProcessor {
public:
    using SortStrategy   = std::function<void(std::vector<int>&)>;
    using FilterStrategy = std::function<bool(int)>;

    void setSortStrategy(SortStrategy s)   { sortStrategy_   = std::move(s); }
    void setFilterStrategy(FilterStrategy f) { filterStrategy_ = std::move(f); }

    std::vector<int> process(std::vector<int> data) const {
        // 应用过滤策略
        if (filterStrategy_) {
            data.erase(std::remove_if(data.begin(), data.end(),
                [this](int x) { return !filterStrategy_(x); }), data.end());
        }
        // 应用排序策略
        if (sortStrategy_) sortStrategy_(data);
        return data;
    }

private:
    SortStrategy   sortStrategy_;
    FilterStrategy filterStrategy_;
};

int main() {
    DataProcessor proc;

    // 策略1：升序 + 只保留偶数
    proc.setSortStrategy([](std::vector<int>& v) { std::sort(v.begin(), v.end()); });
    proc.setFilterStrategy([](int x) { return x % 2 == 0; });

    std::vector<int> data = {5, 3, 8, 1, 6, 2, 9, 4};
    auto result = proc.process(data);
    std::cout << "升序偶数: ";
    for (int x : result) std::cout << x << " ";
    std::cout << "\n";  // 2 4 6 8

    // 策略2：降序 + 只保留大于 4 的值
    proc.setSortStrategy([](std::vector<int>& v) {
        std::sort(v.begin(), v.end(), std::greater<int>{});
    });
    proc.setFilterStrategy([](int x) { return x > 4; });

    result = proc.process(data);
    std::cout << "降序>4: ";
    for (int x : result) std::cout << x << " ";
    std::cout << "\n";  // 9 8 6 5
    return 0;
}
strategy_template.cpp — 编译期策略（模板参数，零开销） 
// 编译期策略：算法作为模板参数，无虚函数开销
struct AscendingSort {
    template<typename It>
    void operator()(It begin, It end) const {
        std::sort(begin, end);
    }
};

struct DescendingSort {
    template<typename It>
    void operator()(It begin, It end) const {
        std::sort(begin, end, std::greater<>{});
    }
};

// Sorter 的策略在编译期确定，内联优化，零额外开销
template<typename SortPolicy = AscendingSort>
class Sorter {
public:
    template<typename Container>
    Container sort(Container data) const {
        policy_(data.begin(), data.end());
        return data;
    }
private:
    SortPolicy policy_;
};

// 使用示例（编译期确定策略）
// Sorter<AscendingSort>{}.sort(data);
// Sorter<DescendingSort>{}.sort(data);

// auto customSort = [](auto b, auto e) { std::sort(b, e, myComp); };

🔄 5. CRTP — 奇异递归模板模式

CRTP（Curiously Recurring Template Pattern）是一种 C++ 惯用法：派生类将自身作为基类的模板参数。这样基类可以在编译期调用派生类的方法，实现静态多态（无虚函数开销）。

crtp.cpp — 静态多态与接口复用 
#include <iostream>
#include <string>
#include <cmath>

// CRTP 基类：提供通用接口，派生类提供实现
template<typename Derived>
class Shape {
public:
    // 通过 CRTP 调用派生类的实现（编译期分发，无虚函数）
    double area() const {
        return static_cast<const Derived*>(this)->areaImpl();
    }

    double perimeter() const {
        return static_cast<const Derived*>(this)->perimeterImpl();
    }

    // 基类提供通用方法，复用派生类接口
    void describe() const {
        std::cout << static_cast<const Derived*>(this)->name()
                  << ": 面积=" << area()
                  << " 周长=" << perimeter() << "\n";
    }

    bool largerThan(const Shape& other) const {
        return area() > other.area();
    }
};

// 派生类：提供具体实现
class Circle : public Shape<Circle> {
public:
    explicit Circle(double r) : radius(r) {}
    double areaImpl()      const { return M_PI * radius * radius; }
    double perimeterImpl() const { return 2 * M_PI * radius; }
    std::string name()     const { return "Circle(r=" + std::to_string(radius) + ")"; }
    double radius;
};

class Rectangle : public Shape<Rectangle> {
public:
    Rectangle(double w, double h) : width(w), height(h) {}
    double areaImpl()      const { return width * height; }
    double perimeterImpl() const { return 2 * (width + height); }
    std::string name()     const { return "Rect(" + std::to_string(width) + "x" + std::to_string(height) + ")"; }
    double width, height;
};

// CRTP 应用2：Mixin 模式 — 为类添加通用能力
template<typename Derived>
class Comparable {
public:
    bool operator!=(const Derived& rhs) const {
        return !(static_cast<const Derived&>(*this) == rhs);
    }
    bool operator>(const Derived& rhs) const  { return rhs < static_cast<const Derived&>(*this); }
    bool operator<=(const Derived& rhs) const { return !(static_cast<const Derived&>(*this) > rhs); }
    bool operator>=(const Derived& rhs) const { return !(static_cast<const Derived&>(*this) < rhs); }
};

// 只需实现 == 和 <，其余运算符由 Comparable 自动提供
class Version : public Comparable<Version> {
public:
    Version(int major, int minor) : major_(major), minor_(minor) {}
    bool operator==(const Version& rhs) const { return major_ == rhs.major_ && minor_ == rhs.minor_; }
    bool operator<(const Version& rhs)  const { return major_ < rhs.major_ || (major_ == rhs.major_ && minor_ < rhs.minor_); }
private:
    int major_, minor_;
};

int main() {
    Circle c(5.0);
    Rectangle r(4.0, 6.0);

    c.describe();   // Circle: 面积=78.54 周长=31.42
    r.describe();   // Rect: 面积=24.00 周长=20.00

    // 编译期分发：无虚函数表查找
    std::cout << "圆比矩形大？" << (c.largerThan(r) ? "是" : "否") << "\n";

    // Comparable Mixin
    Version v1(1, 2), v2(1, 5), v3(2, 0);
    std::cout << "v1 < v2? " << (v1 < v2) << "\n";   // 1
    std::cout << "v2 > v3? " << (v2 > v3) << "\n";   // 0
    std::cout << "v1 != v2? " << (v1 != v2) << "\n"; // 1
    return 0;
}

// 对比 CRTP vs 虚函数：
// 虚函数：运行时分发，每次调用一次间接寻址（~1ns 额外开销）
// CRTP：编译期分发，可以内联，零额外开销
// 代价：CRTP 类型不能放入同一容器（Circle 和 Rectangle 是不同基类实例）

🧪 综合代码示例：插件系统

综合运用工厂模式 + 观察者模式构建一个可扩展的插件系统。

plugin_system.cpp 
#include <iostream>
#include <memory>
#include <string>
#include <vector>
#include <functional>
#include <unordered_map>

// 插件接口
struct IPlugin {
    virtual ~IPlugin() = default;
    virtual std::string name() const = 0;
    virtual void execute(const std::string& input) = 0;
};

// 具体插件
struct UpperCasePlugin : IPlugin {
    std::string name() const override { return "UpperCase"; }
    void execute(const std::string& input) override {
        std::string out = input;
        for (char& c : out) c = std::toupper(c);
        std::cout << "[UpperCase] " << out << "\n";
    }
};

struct ReversePlugin : IPlugin {
    std::string name() const override { return "Reverse"; }
    void execute(const std::string& input) override {
        std::string out(input.rbegin(), input.rend());
        std::cout << "[Reverse] " << out << "\n";
    }
};

// 插件管理器：工厂注册 + 观察者通知
class PluginManager {
public:
    using Factory = std::function<std::unique_ptr<IPlugin>()>;

    // 注册工厂
    void registerPlugin(const std::string& name, Factory factory) {
        factories_[name] = std::move(factory);
        for (auto& cb : onRegistered_) cb(name);
    }

    // 订阅注册事件
    void onPluginRegistered(std::function<void(std::string)> cb) {
        onRegistered_.push_back(std::move(cb));
    }

    // 运行所有插件
    void runAll(const std::string& input) const {
        for (auto& [name, factory] : factories_) {
            auto plugin = factory();
            plugin->execute(input);
        }
    }

private:
    std::unordered_map<std::string, Factory> factories_;
    std::vector<std::function<void(std::string)>> onRegistered_;
};

int main() {
    PluginManager manager;

    // 观察者：监控插件注册
    manager.onPluginRegistered([](const std::string& name) {
        std::cout << "[系统] 插件已注册: " << name << "\n";
    });

    // 工厂注册
    manager.registerPlugin("UpperCase", [] { return std::make_unique<UpperCasePlugin>(); });
    manager.registerPlugin("Reverse",   [] { return std::make_unique<ReversePlugin>();   });

    std::cout << "\n--- 运行所有插件 ---\n";
    manager.runAll("hello world");
    return 0;
}
