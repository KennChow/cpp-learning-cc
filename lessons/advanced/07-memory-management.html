📐 1. 堆与栈 — 内存布局与开销

程序运行时，内存从高地址到低地址大致分为：代码段、全局/静态数据段、堆（向高地址增长）、未映射区、栈（向低地址增长）。理解二者的差异是性能优化的基础。

栈与堆分配对比 
#include <iostream>
#include <chrono>
#include <memory>

// ===== 栈分配：极快，自动管理 =====
void stack_example() {
    int arr[1024];          // 栈上分配：仅移动栈指针，O(1)
    arr[0] = 42;            // 无需释放，函数返回时自动回收
}   // arr 在此销毁

// ===== 堆分配：灵活，但有开销 =====
void heap_example() {
    int* arr = new int[1024];  // 堆分配：系统调用 + 内存管理开销
    arr[0] = 42;
    delete[] arr;              // 必须手动释放，否则内存泄漏
}

// ===== 性能测量：栈 vs 堆 =====
void benchmark() {
    const int N = 1'000'000;
    auto t1 = std::chrono::high_resolution_clock::now();

    // 栈分配（在循环外）
    for (int i = 0; i < N; ++i) {
        int x = i * i;  // 栈上，极快
        (void)x;
    }
    auto t2 = std::chrono::high_resolution_clock::now();

    // 堆分配
    for (int i = 0; i < N; ++i) {
        int* p = new int(i * i);  // 堆分配，约 100ns/次
        delete p;
    }
    auto t3 = std::chrono::high_resolution_clock::now();

    using ms = std::chrono::milliseconds;
    std::cout << "栈: " << std::chrono::duration_cast<ms>(t2-t1).count() << "ms\n";
    std::cout << "堆: " << std::chrono::duration_cast<ms>(t3-t2).count() << "ms\n";
    // 典型结果：堆比栈慢 10~100 倍
}

// ===== 内存布局（典型 x86-64 Linux 进程）=====
// 0xFFFF...  内核空间
// 0x7FFF...  栈（向下增长）
//            ↓ 栈帧
//            ...
//            ↑ 堆（向上增长）
// 0x...      堆
// 0x...      BSS（未初始化全局变量）
// 0x...      数据段（已初始化全局变量）
// 0x...      代码段（只读）

int global_var = 10;          // 数据段
int uninit_global;            // BSS 段（自动初始化为 0）

int main() {
    int stack_var = 20;       // 栈
    int* heap_var = new int(30);  // 堆

    std::cout << "全局变量地址: " << &global_var  << "\n";
    std::cout << "栈变量地址:   " << &stack_var   << "\n";
    std::cout << "堆变量地址:   " << heap_var     << "\n";
    // 典型输出：堆地址 < 栈地址

    delete heap_var;
    benchmark();
    return 0;
}

关键对比：

  • 栈：分配仅移动栈指针，约 1ns；大小受限（默认 1~8 MB）；生命周期与作用域绑定
  • 堆：需要与 OS/分配器交互，约 100ns；大小受限于虚拟内存；生命周期手动管理
  • 缓存友好性：栈内存连续性好，堆内存可能碎片化，影响 CPU cache 命中率

🔧 2. new/delete 内部机制

new 表达式分两步：调用 operator new 分配原始内存，然后在该内存上调用构造函数。delete 同样分两步：调用析构函数，再调用 operator delete 释放内存。

operator new 机制与重载 
#include <iostream>
#include <cstdlib>   // malloc/free
#include <new>       // std::bad_alloc, placement new

// ===== new 表达式的等价展开 =====
// MyClass* p = new MyClass(args...);
// 等价于：
//   void* mem = operator new(sizeof(MyClass));   // 1. 分配内存
//   MyClass* p = new(mem) MyClass(args...);       // 2. 构造（placement new）

// ===== 全局重载 operator new（调试/统计用）=====
static size_t total_allocated = 0;

void* operator new(size_t size) {
    total_allocated += size;
    std::cout << "[alloc] " << size << " bytes (total: " << total_allocated << ")\n";
    void* p = std::malloc(size);
    if (!p) throw std::bad_alloc{};
    return p;
}

void operator delete(void* p) noexcept {
    std::cout << "[free]\n";
    std::free(p);
}

void operator delete(void* p, size_t size) noexcept {
    total_allocated -= size;
    std::free(p);
}

// ===== 类级别重载（只影响该类）=====
class MyClass {
public:
    int data;
    MyClass(int v) : data(v) {}

    // 类级 operator new：优先于全局版本
    static void* operator new(size_t size) {
        std::cout << "MyClass::operator new(" << size << ")\n";
        return ::operator new(size);  // 委托全局版本
    }

    static void operator delete(void* p, size_t size) noexcept {
        std::cout << "MyClass::operator delete\n";
        ::operator delete(p, size);
    }
};

// ===== nothrow new：分配失败返回 nullptr 而非抛异常 =====
void nothrow_example() {
    // 请求极大内存，可能失败
    void* p = operator new(SIZE_MAX, std::nothrow);
    if (!p) {
        std::cout << "内存分配失败（nothrow）\n";
    }
}

int main() {
    // new 分两步：分配 + 构造
    MyClass* obj = new MyClass(42);
    std::cout << "data = " << obj->data << "\n";
    delete obj;

    // new[] 分配数组
    int* arr = new int[5]{1, 2, 3, 4, 5};
    delete[] arr;  // 必须用 delete[]！

    return 0;
}
placement new — 在指定内存上构造对象 
#include <new>
#include <iostream>
#include <memory>    // std::launder (C++17)

struct Widget {
    int id;
    double value;
    Widget(int i, double v) : id(i), value(v) {
        std::cout << "Widget(" << id << ") 构造\n";
    }
    ~Widget() {
        std::cout << "Widget(" << id << ") 析构\n";
    }
};

void placement_new_demo() {
    // 分配原始内存（alignas 保证对齐）
    alignas(Widget) unsigned char buf[sizeof(Widget)];

    // placement new：在 buf 上构造 Widget，不分配新内存
    Widget* w = new(buf) Widget(1, 3.14);
    std::cout << "id=" << w->id << ", value=" << w->value << "\n";

    // 必须显式调用析构函数（不能 delete，因为内存不是 new 分配的）
    w->~Widget();
    // 之后 buf 可重复使用

    // 重新在同一内存上构造另一个对象
    Widget* w2 = new(buf) Widget(2, 2.718);
    w2->~Widget();
}

// placement new 的典型应用：延迟初始化
template<typename T>
class LazyInit {
    alignas(T) unsigned char storage_[sizeof(T)];
    bool initialized_ = false;
public:
    template<typename... Args>
    void init(Args&&... args) {
        if (!initialized_) {
            new(storage_) T(std::forward<Args>(args)...);
            initialized_ = true;
        }
    }

    T& get() {
        // std::launder：告知编译器指针指向了新对象（C++17）
        return *std::launder(reinterpret_cast<T*>(storage_));
    }

    ~LazyInit() {
        if (initialized_) {
            get().~T();
        }
    }
};

int main() {
    placement_new_demo();

    LazyInit<Widget> lazy;
    // 此时 Widget 尚未构造
    lazy.init(42, 1.618);
    std::cout << "lazy id=" << lazy.get().id << "\n";
    return 0;
}

⚙️ 3. 自定义分配器

C++ 标准库容器通过模板参数接受分配器，允许你替换默认的堆分配策略。std::allocator<T> 是默认实现，自定义分配器需满足其接口约定。

std::allocator 接口与自定义 Arena 分配器 
#include <memory>
#include <vector>
#include <iostream>
#include <cassert>

// ===== std::allocator 基本接口 =====
void std_allocator_demo() {
    std::allocator<int> alloc;

    // 分配 5 个 int 的空间（不构造）
    int* p = alloc.allocate(5);

    // 在分配的内存上构造对象（C++20 前用 construct，之后直接 placement new）
    for (int i = 0; i < 5; ++i) {
        std::construct_at(p + i, i * 10);  // C++20
    }

    for (int i = 0; i < 5; ++i) {
        std::cout << p[i] << " ";  // 0 10 20 30 40
    }
    std::cout << "\n";

    // 析构后释放
    std::destroy_n(p, 5);  // C++17
    alloc.deallocate(p, 5);
}

// ===== Arena 分配器：从预分配的大块内存中线性分配 =====
template<size_t N>
class ArenaAllocator {
    alignas(std::max_align_t) char buf_[N];
    char* ptr_ = buf_;
public:
    void* allocate(size_t size, size_t align = alignof(std::max_align_t)) {
        // 对齐当前指针
        char* aligned = reinterpret_cast<char*>(
            (reinterpret_cast<uintptr_t>(ptr_) + align - 1) & ~(align - 1)
        );
        if (aligned + size > buf_ + N) return nullptr;  // 容量不足
        ptr_ = aligned + size;
        return aligned;
    }

    void reset() { ptr_ = buf_; }  // O(1) 释放全部内存！

    size_t used() const { return static_cast<size_t>(ptr_ - buf_); }
    size_t remaining() const { return N - used(); }
};

// ===== STL 兼容的分配器包装 =====
template<typename T, size_t N>
class ArenaSTLAllocator {
    ArenaAllocator<N>* arena_;
public:
    using value_type = T;

    explicit ArenaSTLAllocator(ArenaAllocator<N>& a) : arena_(&a) {}

    template<typename U>
    ArenaSTLAllocator(const ArenaSTLAllocator<U, N>& other) : arena_(other.arena_) {}

    T* allocate(size_t n) {
        void* p = arena_->allocate(n * sizeof(T), alignof(T));
        if (!p) throw std::bad_alloc{};
        return static_cast<T*>(p);
    }

    void deallocate(T*, size_t) noexcept {
        // Arena 分配器不单独释放，reset() 统一回收
    }

    template<typename U> friend class ArenaSTLAllocator;
};

int main() {
    std_allocator_demo();

    // Arena 分配器：所有分配从同一块栈内存中线性切割
    ArenaAllocator<4096> arena;
    ArenaSTLAllocator<int, 4096> alloc(arena);

    std::vector<int, ArenaSTLAllocator<int, 4096>> vec(alloc);
    for (int i = 0; i < 100; ++i) vec.push_back(i);

    std::cout << "Arena 已用: " << arena.used() << " bytes\n";
    arena.reset();  // 瞬间释放所有内存
    std::cout << "reset 后已用: " << arena.used() << " bytes\n";
    return 0;
}

🏊 4. 内存池

内存池（Memory Pool）预先分配大块内存，然后切割为固定大小的块供对象使用。相比每次调用 new/delete，它减少了系统调用次数、消除了碎片化，分配和释放接近 O(1)。

fixed_pool.cpp — 固定大小内存池实现 
#include <iostream>
#include <cstdlib>
#include <cassert>
#include <chrono>

// ===== 固定大小内存池 =====
// 原理：将空闲块用链表串起来，分配时从链表头取，回收时插回链表头
template<size_t BlockSize, size_t BlockCount>
class FixedMemoryPool {
    static_assert(BlockSize >= sizeof(void*), "块大小至少为一个指针");

    // 每个空闲块的头部存放下一个空闲块的指针
    struct FreeBlock { FreeBlock* next; };

    alignas(std::max_align_t) char pool_[BlockSize * BlockCount];
    FreeBlock* free_list_ = nullptr;
    size_t allocated_ = 0;

public:
    FixedMemoryPool() {
        // 初始化：将所有块串成空闲链表
        for (size_t i = 0; i < BlockCount; ++i) {
            auto* blk = reinterpret_cast<FreeBlock*>(pool_ + i * BlockSize);
            blk->next = free_list_;
            free_list_ = blk;
        }
    }

    void* allocate() {
        if (!free_list_) return nullptr;  // 池已满
        FreeBlock* blk = free_list_;
        free_list_ = blk->next;
        ++allocated_;
        return blk;
    }

    void deallocate(void* p) {
        if (!p) return;
        auto* blk = static_cast<FreeBlock*>(p);
        blk->next = free_list_;  // 插回链表头，O(1)
        free_list_ = blk;
        --allocated_;
    }

    size_t allocated() const { return allocated_; }
    size_t remaining() const { return BlockCount - allocated_; }
    bool full() const { return free_list_ == nullptr; }
};

// ===== 带内存池的对象类 =====
struct Particle {
    float x, y, z;
    float vx, vy, vz;
    float mass;

    static FixedMemoryPool<sizeof(Particle), 10000> pool;

    static void* operator new(size_t) {
        void* p = pool.allocate();
        if (!p) throw std::bad_alloc{};
        return p;
    }
    static void operator delete(void* p) noexcept {
        pool.deallocate(p);
    }

    Particle(float x, float y, float z)
        : x(x), y(y), z(z), vx(0), vy(0), vz(0), mass(1.0f) {}
};

// 定义静态成员（类外）
FixedMemoryPool<sizeof(Particle), 10000> Particle::pool;

void benchmark_pool() {
    const int N = 100000;
    std::vector<Particle*> particles;
    particles.reserve(N);

    // 内存池分配
    auto t1 = std::chrono::high_resolution_clock::now();
    for (int i = 0; i < N; ++i) {
        particles.push_back(new Particle(i * 0.1f, i * 0.2f, i * 0.3f));
    }
    for (auto* p : particles) delete p;
    particles.clear();
    auto t2 = std::chrono::high_resolution_clock::now();

    // 普通 new/delete（对比）
    for (int i = 0; i < N; ++i) {
        // 临时禁用内存池，直接用 malloc
        void* mem = std::malloc(sizeof(Particle));
        new(mem) Particle(i * 0.1f, i * 0.2f, i * 0.3f);
        particles.push_back(static_cast<Particle*>(mem));
    }
    for (auto* p : particles) { p->~Particle(); std::free(p); }
    auto t3 = std::chrono::high_resolution_clock::now();

    using us = std::chrono::microseconds;
    std::cout << "内存池: " << std::chrono::duration_cast<us>(t2-t1).count() << " us\n";
    std::cout << "malloc: " << std::chrono::duration_cast<us>(t3-t2).count() << " us\n";
}

int main() {
    // 演示内存池使用
    auto* p1 = new Particle(1.0f, 2.0f, 3.0f);
    auto* p2 = new Particle(4.0f, 5.0f, 6.0f);
    std::cout << "已分配: " << Particle::pool.allocated() << "\n";
    delete p1;
    std::cout << "释放后: " << Particle::pool.allocated() << "\n";
    delete p2;

    benchmark_pool();
    return 0;
}

🔍 5. 内存泄漏检测

内存泄漏是 C++ 中最常见的 bug 之一。现代工具链提供了强大的检测手段：Valgrind（运行时检测）和 AddressSanitizer（编译期插桩，更快）是最常用的两种。

常见内存错误模式 
#include <iostream>
#include <memory>

// ===== 常见内存泄漏模式 =====

// 模式1：忘记 delete
void leak_basic() {
    int* p = new int(42);
    // 忘记 delete p;  → 4 bytes 泄漏
}

// 模式2：异常导致泄漏
void leak_exception() {
    int* p = new int[100];
    // 如果下面抛异常，p 永远不会被释放
    // throw std::runtime_error("oops");
    delete[] p;  // 异常路径上永远到不了这里
}

// 模式3：new[] 与 delete 不匹配（未定义行为）
void mismatch() {
    int* arr = new int[10];
    delete arr;   // 错！应该用 delete[]
}

// 模式4：双重释放（未定义行为）
void double_free() {
    int* p = new int(1);
    delete p;
    // delete p;  // 再次 delete：未定义行为，可能崩溃
}

// 模式5：使用已释放内存（use-after-free）
void use_after_free() {
    int* p = new int(99);
    delete p;
    // std::cout << *p;  // 未定义行为！
}

// ===== 正确做法：RAII + 智能指针 =====
void correct_raii() {
    // unique_ptr：自动释放
    auto p = std::make_unique<int>(42);
    // 异常安全：任何路径离开作用域都会自动释放

    // 数组
    auto arr = std::make_unique<int[]>(100);
    arr[0] = 1;

    // shared_ptr：引用计数
    auto sp1 = std::make_shared<std::string>("hello");
    auto sp2 = sp1;  // 引用计数 = 2
    // 两个都出作用域才释放
}

// ===== ASAN 能检测的错误类型 =====
// 1. heap-buffer-overflow：越界访问
void asan_heap_overflow() {
    int* p = new int[5];
    // p[5] = 1;  // ASAN: heap-buffer-overflow
    delete[] p;
}

// 2. stack-buffer-overflow
void asan_stack_overflow() {
    int arr[5];
    // arr[5] = 1;  // ASAN: stack-buffer-overflow（运行时）
}

// 3. memory leak
void asan_leak() {
    new int(42);  // ASAN: 程序退出时报告 4 bytes leaked
}
检测工具使用方法（Shell 命令） 
# ===== Valgrind：全面但慢（约 10~50x 减速）=====
# 编译：保留调试信息，不优化
g++ -g -O0 my_program.cpp -o my_program

# 运行泄漏检测
valgrind --leak-check=full \
         --show-leak-kinds=all \
         --track-origins=yes \
         --verbose \
         ./my_program

# 典型 Valgrind 输出：
# ==12345== LEAK SUMMARY:
# ==12345==    definitely lost: 4 bytes in 1 blocks
# ==12345==    indirectly lost: 0 bytes in 0 blocks
# ==12345==      possibly lost: 0 bytes in 0 blocks
# ==12345==    still reachable: 0 bytes in 0 blocks


# ===== AddressSanitizer：快（约 2x 减速），GCC/Clang 内置 =====
# 编译时开启 ASAN
g++ -g -fsanitize=address -fno-omit-frame-pointer my_program.cpp -o my_program_asan

# 直接运行，ASAN 自动报告（无需额外命令）
./my_program_asan

# 典型 ASAN 输出：
# =================================================================
# ==12345==ERROR: AddressSanitizer: heap-buffer-overflow
#  on address 0x602000000014 at ...
# READ of size 4 at 0x602000000014
#     #0 0x... in main my_program.cpp:42

# ===== 同时启用 LeakSanitizer（LSAN，通常随 ASAN 自动启用）=====
ASAN_OPTIONS=detect_leaks=1 ./my_program_asan

# ===== MemorySanitizer：检测未初始化内存读（需 Clang）=====
clang++ -g -fsanitize=memory -fno-omit-frame-pointer my_program.cpp -o my_program_msan
./my_program_msan

# ===== UndefinedBehaviorSanitizer：检测 UB =====
g++ -g -fsanitize=undefined my_program.cpp -o my_program_ubsan
./my_program_ubsan

工具选择建议：

  • 日常开发：首选 ASAN（-fsanitize=address），开销小、信息详细
  • 深度排查：Valgrind Memcheck，能检测 ASAN 遗漏的某些问题
  • 未初始化内存：MSan（-fsanitize=memory，需 Clang）
  • 根本解决方案：用智能指针 + RAII，让泄漏不可能发生

🧪 综合代码示例：对象池 + RAII 包装

object_pool.cpp 
#include <iostream>
#include <memory>
#include <vector>
#include <functional>

// ===== 通用对象池（线程不安全版，演示用）=====
template<typename T>
class ObjectPool {
    std::vector<std::unique_ptr<T>> pool_;
    std::vector<T*> free_list_;

public:
    explicit ObjectPool(size_t initial = 16) {
        reserve(initial);
    }

    void reserve(size_t n) {
        pool_.reserve(pool_.size() + n);
        for (size_t i = 0; i < n; ++i) {
            pool_.push_back(std::make_unique<T>());
            free_list_.push_back(pool_.back().get());
        }
    }

    // 返回 RAII 包装的对象，归还时自动入池
    struct Handle {
        T* ptr;
        ObjectPool* pool;

        T* operator->() { return ptr; }
        T& operator*()  { return *ptr; }

        ~Handle() {
            if (ptr && pool) {
                pool->free_list_.push_back(ptr);
                ptr = nullptr;
            }
        }

        // 移动语义
        Handle(Handle&& o) noexcept : ptr(o.ptr), pool(o.pool) {
            o.ptr = nullptr; o.pool = nullptr;
        }
        Handle& operator=(Handle&& o) noexcept {
            if (this != &o) {
                if (ptr && pool) pool->free_list_.push_back(ptr);
                ptr = o.ptr; pool = o.pool;
                o.ptr = nullptr; o.pool = nullptr;
            }
            return *this;
        }
        Handle(const Handle&) = delete;
        Handle& operator=(const Handle&) = delete;
    private:
        friend class ObjectPool;
        Handle(T* p, ObjectPool* pool) : ptr(p), pool(pool) {}
    };

    Handle acquire() {
        if (free_list_.empty()) reserve(pool_.size());  // 自动扩容
        T* obj = free_list_.back();
        free_list_.pop_back();
        return Handle{obj, this};
    }

    size_t available() const { return free_list_.size(); }
    size_t total() const     { return pool_.size(); }
};

// ===== 演示 =====
struct Connection {
    int id = 0;
    bool active = false;

    void open(int conn_id)  { id = conn_id; active = true;  }
    void close()            { active = false; }
};

int main() {
    ObjectPool<Connection> conn_pool(4);
    std::cout << "初始可用: " << conn_pool.available() << "\n";

    {
        auto c1 = conn_pool.acquire();
        c1->open(1001);
        auto c2 = conn_pool.acquire();
        c2->open(1002);

        std::cout << "使用中可用: " << conn_pool.available() << "\n";
        std::cout << "c1 id=" << c1->id << ", active=" << c1->active << "\n";
        // c1, c2 析构时自动归还池
    }

    std::cout << "归还后可用: " << conn_pool.available() << "\n";
    std::cout << "总对象数:   " << conn_pool.total() << "\n";
    return 0;
}
