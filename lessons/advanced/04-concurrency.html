🧵 1. std::thread 创建线程

线程创建与管理 
#include <thread>
#include <iostream>

void worker(int id, int n) {
    for (int i = 0; i < n; ++i) {
        std::cout << "Thread " << id << ": " << i << "\n";
    }
}

int main() {
    // 创建线程（立即开始执行）
    std::thread t1(worker, 1, 5);  // 函数 + 参数
    std::thread t2([](){ std::cout << "Lambda thread\n"; });

    // join()：等待线程结束（阻塞）
    t1.join();
    t2.join();

    // detach()：线程独立运行，不再等待
    std::thread t3(worker, 3, 3);
    t3.detach();  // t3 在后台运行，main 不等待
    // 注意：main 结束后 detach 的线程会被强制终止

    // 获取线程 ID
    std::thread t4([](){
        std::cout << "Thread ID: " << std::this_thread::get_id() << "\n";
    });
    std::cout << "t4 ID: " << t4.get_id() << "\n";
    t4.join();

    // 硬件并发数
    unsigned int cores = std::thread::hardware_concurrency();
    std::cout << "CPU 核心数: " << cores << "\n";

    return 0;
    // 注意：若线程对象销毁时既未 join 也未 detach，程序会 terminate！
}

⚠️ 2. 数据竞争（Data Race）

当多个线程同时访问同一变量，且至少一个是写操作，而没有同步机制时，就发生数据竞争，结果是未定义行为（UB）。

数据竞争示例（危险！） 
int counter = 0;  // 共享变量

void unsafe_increment() {
    for (int i = 0; i < 100000; ++i) {
        ++counter;  // 非原子操作！
        // 实际是：读 counter → 加 1 → 写回
        // 线程切换可能在任何步骤之间发生
    }
}

int main() {
    std::thread t1(unsafe_increment);
    std::thread t2(unsafe_increment);
    t1.join();
    t2.join();
    // counter 的结果不是 200000，而是随机的！
    std::cout << counter << "\n";  // 可能是 123456，每次不同
    return 0;
}

🔐 3. mutex 与 lock_guard

互斥锁保护共享数据 
#include <mutex>

std::mutex mtx;
int counter = 0;

void safe_increment() {
    for (int i = 0; i < 100000; ++i) {
        // lock_guard：RAII 风格，构造时加锁，析构时解锁
        std::lock_guard<std::mutex> lock(mtx);
        ++counter;  // 临界区：同一时刻只有一个线程能执行
        // lock 在此行结束时析构，自动解锁
    }
}

// unique_lock：更灵活（可以手动解锁、配合 condition_variable）
void flexible_lock() {
    std::unique_lock<std::mutex> lock(mtx);
    // ... 执行一些操作 ...
    lock.unlock();  // 提前解锁

    // ... 不需要保护的操作 ...

    lock.lock();    // 重新加锁
    // ... 继续临界区 ...
}  // lock 析构时若已加锁则解锁

// 避免死锁：同时锁多个 mutex
std::mutex mtx1, mtx2;
void safe_multi_lock() {
    // C++17：std::scoped_lock（推荐）
    std::scoped_lock lock(mtx1, mtx2);  // 原子地锁定多个 mutex

    // 等价的 C++11 方式
    // std::lock(mtx1, mtx2);  // 死锁安全的同时锁定
    // std::lock_guard<std::mutex> l1(mtx1, std::adopt_lock);
    // std::lock_guard<std::mutex> l2(mtx2, std::adopt_lock);
}

📡 4. condition_variable — 线程间通信

生产者-消费者模式 
#include <condition_variable>
#include <queue>

std::queue<int> buffer;
std::mutex mtx;
std::condition_variable cv;
bool done = false;

void producer() {
    for (int i = 0; i < 10; ++i) {
        {
            std::lock_guard<std::mutex> lock(mtx);
            buffer.push(i);
            std::cout << "生产: " << i << "\n";
        }
        cv.notify_one();  // 通知一个等待的线程
    }
    {
        std::lock_guard<std::mutex> lock(mtx);
        done = true;
    }
    cv.notify_all();  // 通知所有等待的线程
}

void consumer(int id) {
    while (true) {
        std::unique_lock<std::mutex> lock(mtx);
        // wait：释放锁并等待，被唤醒后重新获取锁
        // 带谓词版本：防止虚假唤醒（spurious wakeup）
        cv.wait(lock, []{ return !buffer.empty() || done; });

        if (buffer.empty() && done) break;

        int val = buffer.front();
        buffer.pop();
        lock.unlock();  // 处理前释放锁

        std::cout << "消费者 " << id << " 消费: " << val << "\n";
    }
}

int main() {
    std::thread prod(producer);
    std::thread cons1(consumer, 1);
    std::thread cons2(consumer, 2);
    prod.join(); cons1.join(); cons2.join();
    return 0;
}

🔮 5. std::async 与 std::future

异步任务 
#include <future>

// 计算密集型任务
long long heavy_computation(int n) {
    long long sum = 0;
    for (int i = 0; i < n; ++i) sum += i;
    return sum;
}

int main() {
    // std::async：在新线程中异步执行任务
    auto future1 = std::async(std::launch::async,
        heavy_computation, 1000000);

    // 主线程可以同时做其他事
    std::cout << "异步任务已启动，主线程继续工作...\n";
    std::this_thread::sleep_for(std::chrono::milliseconds(10));

    // get()：阻塞等待并获取结果
    long long result = future1.get();
    std::cout << "结果: " << result << "\n";

    // 并行执行多个任务
    auto f1 = std::async(std::launch::async, heavy_computation, 1000000);
    auto f2 = std::async(std::launch::async, heavy_computation, 2000000);
    auto f3 = std::async(std::launch::async, heavy_computation, 3000000);

    std::cout << "三个结果: "
              << f1.get() << ", "
              << f2.get() << ", "
              << f3.get() << "\n";

    // promise + future：手动控制
    std::promise<int> prom;
    std::future<int> fut = prom.get_future();

    std::thread t([&prom](){
        std::this_thread::sleep_for(std::chrono::milliseconds(100));
        prom.set_value(42);  // 设置值，唤醒等待的 get()
    });

    std::cout << "等待 promise: " << fut.get() << "\n";  // 42
    t.join();

    // std::launch::deferred：延迟执行（调用 get() 时才在当前线程执行）
    auto lazy = std::async(std::launch::deferred, heavy_computation, 500);
    std::cout << "延迟执行结果: " << lazy.get() << "\n";

    return 0;
}

⚛️ 6. std::atomic — 原子操作

原子类型 
#include <atomic>

// atomic 保证操作的原子性（无需 mutex）
std::atomic<int> atomic_counter{0};

void atomic_increment() {
    for (int i = 0; i < 100000; ++i) {
        ++atomic_counter;  // 原子操作，线程安全
        // 等价于：atomic_counter.fetch_add(1)
    }
}

// atomic 常用操作
std::atomic<int> val{10};
val.store(20);             // 原子写入
int v = val.load();        // 原子读取
int old = val.fetch_add(5); // 原子加，返回旧值
int expected = 20;
val.compare_exchange_strong(expected, 100);
// 如果 val == expected，则 val = 100，返回 true
// 否则 expected = val 的当前值，返回 false

// 原子标志（最轻量的原子类型）
std::atomic_flag flag = ATOMIC_FLAG_INIT;
// test_and_set：原子地检查并置位
if (!flag.test_and_set()) {
    std::cout << "第一次进入\n";
}
flag.clear();  // 清除标志

// atomic vs mutex 的选择：
// - atomic：适合简单的计数器、标志位
// - mutex：适合保护复杂的数据结构或多步操作

🧪 综合代码示例：线程池

simple_thread_pool.cpp 
#include <iostream>
#include <thread>
#include <mutex>
#include <condition_variable>
#include <queue>
#include <functional>
#include <vector>
#include <atomic>

class ThreadPool {
    std::vector<std::thread> workers_;
    std::queue<std::function<void()>> tasks_;
    std::mutex queue_mutex_;
    std::condition_variable cv_;
    std::atomic<bool> stop_{false};

public:
    ThreadPool(size_t num_threads) {
        for (size_t i = 0; i < num_threads; ++i) {
            workers_.emplace_back([this]() {
                while (true) {
                    std::function<void()> task;
                    {
                        std::unique_lock<std::mutex> lock(queue_mutex_);
                        cv_.wait(lock, [this]{
                            return stop_ || !tasks_.empty();
                        });
                        if (stop_ && tasks_.empty()) return;
                        task = std::move(tasks_.front());
                        tasks_.pop();
                    }
                    task();  // 执行任务
                }
            });
        }
    }

    template<typename F>
    void enqueue(F&& f) {
        {
            std::lock_guard<std::mutex> lock(queue_mutex_);
            tasks_.emplace(std::forward<F>(f));
        }
        cv_.notify_one();
    }

    ~ThreadPool() {
        stop_ = true;
        cv_.notify_all();
        for (auto& t : workers_) t.join();
    }
};

int main() {
    ThreadPool pool(4);  // 4 个工作线程
    std::mutex cout_mtx;

    for (int i = 0; i < 10; ++i) {
        pool.enqueue([i, &cout_mtx]() {
            std::this_thread::sleep_for(std::chrono::milliseconds(50));
            std::lock_guard<std::mutex> lock(cout_mtx);
            std::cout << "任务 " << i << " 完成，线程: "
                      << std::this_thread::get_id() << "\n";
        });
    }
    // ThreadPool 析构时等待所有任务完成
    return 0;
}
