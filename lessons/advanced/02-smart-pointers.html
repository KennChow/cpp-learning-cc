❓ 1. 为什么需要智能指针

裸指针（raw pointer）的问题在于需要手动管理内存，容易出现内存泄漏、悬空指针、重复释放等问题：

裸指针的问题 
// 危险的裸指针代码
void risky_function() {
    MyClass* obj = new MyClass();

    if (some_condition()) {
        return;  // 内存泄漏！obj 未被释放
    }

    process(obj);  // 如果 process 抛出异常呢？

    delete obj;    // 只有正常执行才能到这里
}

// RAII 思想：资源获取即初始化
// 智能指针在析构时自动释放内存，无论以何种方式离开作用域
void safe_function() {
    auto obj = std::make_unique<MyClass>();  // 自动管理

    if (some_condition()) {
        return;  // OK！unique_ptr 析构时自动 delete
    }

    process(*obj);  // 如果抛出异常，unique_ptr 析构处理
}  // 作用域结束，自动释放

🔒 2. unique_ptr — 独占所有权

std::unique_ptr 表示对资源的独占所有权。同一时刻只有一个 unique_ptr 可以持有某个资源。它不可拷贝，但可以移动。

unique_ptr 使用 
#include <memory>

// 创建 unique_ptr（优先用 make_unique）
auto p1 = std::make_unique<int>(42);
auto p2 = std::make_unique<std::string>("hello");

// 解引用和访问
std::cout << *p1 << "\n";        // 42
std::cout << p2->size() << "\n"; // 5

// 移动所有权（p1 变为 nullptr）
auto p3 = std::move(p1);
std::cout << (p1 == nullptr) << "\n"; // 1（true）
std::cout << *p3 << "\n";             // 42

// 不能拷贝！
// auto p4 = p3;  // 编译错误：拷贝被删除

// 释放所有权（返回裸指针，不再管理）
int* raw = p3.release();  // p3 变为 nullptr
delete raw;               // 需要手动释放

// 重置
p3.reset(new int(100));   // 释放旧资源，持有新资源
p3.reset();               // 释放资源，变为 nullptr

// 自定义删除器
auto file_ptr = std::unique_ptr<FILE, decltype(&fclose)>(
    fopen("test.txt", "r"), &fclose
);  // 超出作用域时自动调用 fclose

// 管理数组
auto arr = std::make_unique<int[]>(10);
arr[0] = 1; arr[5] = 5;  // 使用下标访问

函数参数中的 unique_ptr

void f(unique_ptr<T> p) — 接管所有权（调用者必须 std::move）

void f(unique_ptr<T>& p) — 可能修改 unique_ptr 本身（如 reset）

void f(T* p) 或 void f(T& p) — 只是借用，不转移所有权（推荐）

🤝 3. shared_ptr — 共享所有权

std::shared_ptr 允许多个指针共享同一资源，通过引用计数跟踪有多少个 shared_ptr 指向该资源。当最后一个持有者销毁时，资源被释放。

shared_ptr 使用 
// 创建 shared_ptr
auto sp1 = std::make_shared<MyClass>("data");
std::cout << sp1.use_count() << "\n";  // 1

{
    auto sp2 = sp1;  // 拷贝：引用计数 +1
    std::cout << sp1.use_count() << "\n";  // 2
    std::cout << sp2.use_count() << "\n";  // 2（同一控制块）

    auto sp3 = sp1;  // 再次拷贝
    std::cout << sp1.use_count() << "\n";  // 3
}  // sp2, sp3 销毁，引用计数 -2

std::cout << sp1.use_count() << "\n";  // 1

// 为什么优先用 make_shared？
// 1. 一次内存分配（对象 + 控制块放在一起）
auto good = std::make_shared<int>(42);

// 2. 两次内存分配（对象和控制块分开）
std::shared_ptr<int> bad(new int(42));

// 3. 异常安全
// f(std::shared_ptr<T>(new T), g());  // 潜在泄漏
// f(std::make_shared<T>(), g());       // 安全

👻 4. weak_ptr — 弱引用与循环引用解决

std::weak_ptr 是对 shared_ptr 管理对象的非拥有性引用。它不增加引用计数，主要用于解决循环引用问题。

循环引用问题与 weak_ptr 解决方案 
// 问题：循环引用导致内存泄漏
struct NodeBad {
    std::shared_ptr<NodeBad> next;  // 强引用
    ~NodeBad() { std::cout << "NodeBad 析构\n"; }
};

{
    auto n1 = std::make_shared<NodeBad>();
    auto n2 = std::make_shared<NodeBad>();
    n1->next = n2;  // n2 引用计数: 2
    n2->next = n1;  // n1 引用计数: 2
    // 离开作用域：n1/n2 计数 -1 变为 1，不为0，无法释放！
    // 内存泄漏！析构函数不会被调用
}

// 解决方案：用 weak_ptr 打破循环
struct NodeGood {
    std::shared_ptr<NodeGood> next;   // 强引用（子持有父用 weak_ptr）
    std::weak_ptr<NodeGood> parent;   // 弱引用，不增加计数
    std::string name;
    ~NodeGood() { std::cout << name << " 析构\n"; }
};

{
    auto root = std::make_shared<NodeGood>();
    root->name = "root";
    auto child = std::make_shared<NodeGood>();
    child->name = "child";

    root->next = child;      // root 拥有 child
    child->parent = root;    // child 只弱引用 root（不增加 root 计数）
    // root 引用计数始终为 1
}  // 正常析构：先 child，后 root

// 使用 weak_ptr 访问资源
void use_weak_ptr(std::weak_ptr<NodeGood> wp) {
    // 尝试获取 shared_ptr（可能失败，如果对象已被销毁）
    if (auto sp = wp.lock()) {  // lock() 返回 shared_ptr
        std::cout << "对象仍然存在: " << sp->name << "\n";
    } else {
        std::cout << "对象已被销毁\n";
    }
}

// weak_ptr 还可以检查是否过期
std::weak_ptr<int> wp;
{
    auto sp = std::make_shared<int>(42);
    wp = sp;
    std::cout << wp.expired() << "\n";  // 0（未过期）
}
std::cout << wp.expired() << "\n";  // 1（已过期）

🧪 综合代码示例

smart_pointers_demo.cpp 
#include <iostream>
#include <memory>
#include <vector>
#include <string>

struct Resource {
    std::string name;
    Resource(std::string n) : name(std::move(n)) {
        std::cout << "创建: " << name << "\n";
    }
    ~Resource() { std::cout << "销毁: " << name << "\n"; }
};

// unique_ptr：工厂函数
std::unique_ptr<Resource> create_resource(const std::string& name) {
    return std::make_unique<Resource>(name);
}

// 接受 unique_ptr（转移所有权）
void take_ownership(std::unique_ptr<Resource> res) {
    std::cout << "接管: " << res->name << "\n";
}  // res 在此析构

// 借用（不转移所有权）
void borrow(const Resource& res) {
    std::cout << "借用: " << res.name << "\n";
}

// shared_ptr：多个所有者
void demo_shared() {
    std::cout << "\n--- shared_ptr ---\n";
    auto sp1 = std::make_shared<Resource>("共享资源");
    std::cout << "计数: " << sp1.use_count() << "\n"; // 1

    auto sp2 = sp1;  // 共享
    std::cout << "计数: " << sp1.use_count() << "\n"; // 2

    std::vector<std::shared_ptr<Resource>> vec;
    vec.push_back(sp1);
    std::cout << "计数: " << sp1.use_count() << "\n"; // 3

    vec.clear();
    std::cout << "计数: " << sp1.use_count() << "\n"; // 2
}  // sp2, sp1 销毁，计数→0，资源释放

// weak_ptr：缓存场景
std::weak_ptr<Resource> cache;

std::shared_ptr<Resource> get_cached() {
    if (auto sp = cache.lock()) {
        std::cout << "缓存命中\n";
        return sp;
    }
    std::cout << "缓存未命中，创建新对象\n";
    auto sp = std::make_shared<Resource>("缓存对象");
    cache = sp;
    return sp;
}

int main() {
    std::cout << "--- unique_ptr ---\n";
    auto r = create_resource("独占资源");
    borrow(*r);  // 借用，不转移
    take_ownership(std::move(r));  // 转移所有权
    // r 现在为 nullptr

    demo_shared();

    std::cout << "\n--- weak_ptr 缓存 ---\n";
    {
        auto sp = get_cached();  // 创建新对象
        auto sp2 = get_cached(); // 缓存命中
        std::cout << "use_count: " << sp.use_count() << "\n"; // 2
    }  // sp, sp2 销毁

    auto sp3 = get_cached();  // 缓存已失效，重新创建

    return 0;
}
