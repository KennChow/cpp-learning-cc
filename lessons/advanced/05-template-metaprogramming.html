⚙️ 1. constexpr — 编译期计算

模板元编程（TMP）的核心思想是将计算从运行时移到编译时，生成零运行时开销的代码。constexpr 是现代 C++ 中实现编译期计算的主要工具。

constexpr 编译期计算 
// constexpr 函数：可在编译期执行
constexpr int factorial(int n) {
    return n <= 1 ? 1 : n * factorial(n - 1);
}

// 编译期验证（static_assert 在编译期检查）
static_assert(factorial(5) == 120, "factorial(5) must be 120");
static_assert(factorial(0) == 1, "factorial(0) must be 1");

// 运行时也可以用（如果参数不是常量）
int runtime_n;
std::cin >> runtime_n;
std::cout << factorial(runtime_n) << "\n";  // 运行时计算

// constexpr 变量：编译期常量
constexpr double PI = 3.14159265358979;
constexpr int BUFFER_SIZE = 1024 * 1024;  // 1MB

// C++14：constexpr 函数可以有更复杂的控制流
constexpr int fibonacci(int n) {
    if (n <= 1) return n;
    int a = 0, b = 1;
    for (int i = 2; i <= n; ++i) {
        int tmp = a + b;
        a = b;
        b = tmp;
    }
    return b;
}
static_assert(fibonacci(10) == 55);

// consteval（C++20）：只能在编译期执行
consteval int square(int n) { return n * n; }
constexpr int s = square(5);  // OK
// int x = 5; int y = square(x);  // 错误！x 不是编译期常量

🔧 2. 模板特化

全特化与偏特化 
// 主模板
template<typename T>
struct TypeInfo {
    static const char* name() { return "unknown"; }
};

// 全特化：为特定类型提供实现
template<>
struct TypeInfo<int> {
    static const char* name() { return "int"; }
};

template<>
struct TypeInfo<double> {
    static const char* name() { return "double"; }
};

std::cout << TypeInfo<int>::name() << "\n";    // "int"
std::cout << TypeInfo<float>::name() << "\n";  // "unknown"

// 偏特化：为一类类型提供实现
template<typename T>
struct TypeInfo<T*> {  // 所有指针类型
    static const char* name() { return "pointer"; }
};

template<typename T>
struct TypeInfo<std::vector<T>> {  // vector 的特化
    static const char* name() { return "vector"; }
};

// 编译期值计算（传统 TMP 风格）
template<int N>
struct Factorial {
    static constexpr int value = N * Factorial<N-1>::value;
};

template<>
struct Factorial<0> {
    static constexpr int value = 1;
};

static_assert(Factorial<5>::value == 120);

🔍 3. SFINAE 与 enable_if

SFINAE（Substitution Failure Is Not An Error）：当编译器尝试用具体类型替换模板参数时，如果某个特化版本替换失败，编译器不报错，而是继续尝试其他候选。这让我们可以根据类型特征选择不同的函数重载。

SFINAE 与 enable_if 
#include <type_traits>

// 只对整数类型有效的函数
template<typename T,
         typename = std::enable_if_t<std::is_integral_v<T>>>
void print_if_int(T val) {
    std::cout << val << " 是整数类型\n";
}

// 只对浮点类型有效的版本
template<typename T,
         typename = std::enable_if_t<std::is_floating_point_v<T>>>
void print_if_float(T val) {
    std::cout << val << " 是浮点类型\n";
}

print_if_int(42);      // OK
// print_if_int(3.14); // 编译错误：SFINAE 排除，无匹配

// 通过返回类型 SFINAE
template<typename T>
std::enable_if_t<std::is_integral_v<T>, T>
safe_double(T x) { return x * 2; }  // 只对整数返回 T

// 通过 void_t 检测类型是否有特定成员
template<typename, typename = void>
struct has_size_method : std::false_type {};

template<typename T>
struct has_size_method<T, std::void_t<decltype(std::declval<T>().size())>>
    : std::true_type {};

static_assert(has_size_method<std::vector<int>>::value);  // true
static_assert(!has_size_method<int>::value);               // false

🌿 4. if constexpr — 编译期分支（C++17）

if constexpr 在编译期选择分支，未选择的分支不会实例化，这解决了 SFINAE 很多场景，代码更清晰。

if constexpr 用法 
// 根据类型选择不同实现
template<typename T>
void process(T val) {
    if constexpr (std::is_integral_v<T>) {
        std::cout << "整数: " << val << " (二进制: " << std::bitset<8>(val) << ")\n";
    } else if constexpr (std::is_floating_point_v<T>) {
        std::cout << std::fixed << std::setprecision(2) << "浮点: " << val << "\n";
    } else if constexpr (std::is_same_v<T, std::string>) {
        std::cout << "字符串: " << val << " (长度: " << val.size() << ")\n";
    } else {
        std::cout << "未知类型\n";
    }
}

process(42);          // 整数
process(3.14);        // 浮点
process(std::string("hi")); // 字符串

// 编译期递归终止
template<typename T>
auto sum(T first) { return first; }

template<typename T, typename... Rest>
auto sum(T first, Rest... rest) {
    return first + sum(rest...);
}

// 或使用折叠表达式（C++17，更简洁）
template<typename... Args>
auto fold_sum(Args... args) { return (... + args); }  // 左折叠

std::cout << fold_sum(1, 2, 3, 4, 5) << "\n";       // 15
std::cout << fold_sum(1.0, 2.5, 3.7) << "\n";       // 7.2

🔬 5. 类型萃取（Type Traits）

常用 type traits 
#include <type_traits>

// 类型查询
static_assert(std::is_integral_v<int>);
static_assert(std::is_floating_point_v<double>);
static_assert(std::is_pointer_v<int*>);
static_assert(std::is_reference_v<int&>);
static_assert(std::is_const_v<const int>);
static_assert(std::is_same_v<int, int>);
static_assert(!std::is_same_v<int, long>);

// 继承关系
static_assert(std::is_base_of_v<std::exception, std::runtime_error>);

// 类型变换
using T1 = std::remove_reference_t<int&>;  // int
using T2 = std::remove_const_t<const int>;  // int
using T3 = std::add_pointer_t<int>;         // int*
using T4 = std::decay_t<int[5]>;            // int*（数组退化）
using T5 = std::conditional_t<true, int, double>;  // int

// 自定义 type trait
template<typename T>
struct is_string : std::false_type {};

template<>
struct is_string<std::string> : std::true_type {};

template<>
struct is_string<const char*> : std::true_type {};

static_assert(is_string<std::string>::value);
static_assert(!is_string<int>::value);

// 类型列表操作（简单示例）
template<typename T>
using RemoveCVRef = std::remove_cv_t<std::remove_reference_t<T>>;

💡 6. Concepts — 约束模板（C++20）

Concepts 是 C++20 引入的革命性特性，让模板约束的表达变得清晰、可读，并能提供更好的错误信息。

Concepts 定义与使用 
// 定义 Concept
template<typename T>
concept Numeric = std::is_arithmetic_v<T>;

template<typename T>
concept Printable = requires(T t) {
    { std::cout << t } -> std::same_as<std::ostream&>;
};

template<typename T>
concept Container = requires(T c) {
    c.begin();
    c.end();
    c.size();
    typename T::value_type;
};

// 使用 Concept 约束模板
template<Numeric T>
T add(T a, T b) { return a + b; }

// 等价写法（requires 子句）
template<typename T> requires Numeric<T>
T multiply(T a, T b) { return a * b; }

// 更复杂的 Concept
template<typename T>
concept Sortable = Container<T> && requires(T c) {
    std::sort(c.begin(), c.end());
};

// 标准库预定义 Concepts（C++20）
#include <concepts>

template<std::integral T>     void int_only(T x) {}
template<std::floating_point T> void float_only(T x) {}
template<std::same_as<int> T>  void int_exact(T x) {}

// Concept + auto 缩写语法
void process(Numeric auto val) {
    std::cout << val * 2 << "\n";
}

// 对比：SFINAE vs Concepts 的错误信息
// SFINAE 错误：模糊的"substitution failure"
// Concepts 错误："T does not satisfy Numeric"（清晰！）

🧪 综合代码示例：编译期斐波那契

tmp_fibonacci.cpp 
#include <iostream>
#include <type_traits>
#include <array>

// 方法1：传统 TMP（模板特化递归）
template<int N>
struct Fib {
    static constexpr long long value = Fib<N-1>::value + Fib<N-2>::value;
};
template<> struct Fib<0> { static constexpr long long value = 0; };
template<> struct Fib<1> { static constexpr long long value = 1; };

// 方法2：constexpr 函数（C++14，更清晰）
constexpr long long fib(int n) {
    if (n <= 1) return n;
    long long a = 0, b = 1;
    for (int i = 2; i <= n; ++i) {
        long long t = a + b; a = b; b = t;
    }
    return b;
}

// 方法3：编译期生成查找表
template<size_t N>
constexpr std::array<long long, N> make_fib_table() {
    std::array<long long, N> arr{};
    if constexpr (N > 0) arr[0] = 0;
    if constexpr (N > 1) arr[1] = 1;
    for (size_t i = 2; i < N; ++i)
        arr[i] = arr[i-1] + arr[i-2];
    return arr;
}

// 类型萃取工具：判断是否为编译期 Fibonacci 数
template<long long N, int Max = 50>
struct is_fibonacci {
    static constexpr bool value = [](){
        long long a = 0, b = 1;
        while (b < N) { long long t = a + b; a = b; b = t; }
        return b == N || N == 0;
    }();
};

int main() {
    // 编译期验证
    static_assert(Fib<10>::value == 55);
    static_assert(fib(20) == 6765);

    // 编译期查找表
    constexpr auto fib_table = make_fib_table<20>();
    static_assert(fib_table[10] == 55);

    // 运行时输出
    std::cout << "传统 TMP — Fib<15> = " << Fib<15>::value << "\n";
    std::cout << "constexpr — fib(15) = " << fib(15) << "\n";

    std::cout << "前 15 个 Fibonacci 数: ";
    for (int i = 0; i < 15; ++i)
        std::cout << fib_table[i] << " ";
    std::cout << "\n";

    // 类型萃取示例
    using T = std::conditional_t<fib(10) > 50, double, int>;
    std::cout << "fib(10)=" << fib(10) << " 选择类型: "
              << (std::is_same_v<T, double> ? "double" : "int") << "\n";

    return 0;
}
// 所有 static_assert 在编译时验证，运行时无额外开销
